}

function ensureMarketHeaders_() {
  const sheet = getOrCreateSheet_(STOCK_CFG.SHEETS.MARKET);

  if (sheet.getLastRow() < 1 || cleanString_(sheet.getRange(1, 1).getValue()) === '') {
    sheet.getRange(1, 1, 1, STOCK_CFG.HEADERS.MARKET.length)
      .setValues([STOCK_CFG.HEADERS.MARKET])
      .setFontWeight('bold');
  }
}


function getExistingTickers_(sheet) {
  const map = {};
  if (sheet.getLastRow() < 2) return map;

  const values = sheet.getRange(2, 1, sheet.getLastRow() - 1, 1).getValues();
  values.forEach(function(row) {
    const ticker = cleanString_(row[0]).toUpperCase();
    if (ticker) map[ticker] = true;
  });

  return map;
}


/*******************************************************
 BUY / SELL
*******************************************************/

function processStockTrade() {
  const lock = LockService.getScriptLock();

  try {
    lock.waitLock(1000);

    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const tradeSheet = ss.getSheetByName(STOCK_CFG.SHEETS.TRADE);

    if (!tradeSheet) {
      SpreadsheetApp.getUi().alert('Stock_Trade sheet not found.');
      return;
    }

    if (!isMarketOpen_()) {
      writeStockTradeResult_('Trade denied: market is closed. Trading hours are 8:00 AM–5:00 PM KST.');
      return;
    }

    SpreadsheetApp.flush();
    tradeSheet.getRange('B8').activate();
    SpreadsheetApp.flush();
    Utilities.sleep(150);

    writeStockTradeResult_('Processing trade...');

    const inputs = tradeSheet.getRange('B3:B6').getDisplayValues();

    const cardId = normalizeCardId_(inputs[0][0]);
    const action = cleanString_(inputs[1][0]).toUpperCase();
    const ticker = cleanString_(inputs[2][0]).toUpperCase();
    const shares = Number(cleanString_(inputs[3][0])) || 0;

    if (!cardId || !action || !ticker || shares <= 0 || !Number.isInteger(shares)) {
      writeStockTradeResult_('Trade failed: Card ID, BUY/SELL, ticker, and whole-number shares are required.');
      return;
    }

    if (action !== 'BUY' && action !== 'SELL') {
      writeStockTradeResult_('Trade failed: Action must be BUY or SELL.');
      return;
    }

    const student = findStudentByCard_(cardId);

    if (!student) {
      writeStockTradeResult_('Trade failed: Card ID not found in master Students sheet: ' + cardId);
      return;
    }

    const stock = findStockByTicker_(ticker);

    if (!stock || !stock.active) {
      writeStockTradeResult_('Trade failed: ticker not found or inactive: ' + ticker);
      return;
    }

    if (action === 'BUY') {
      buyStockShares_(student, stock, shares, tradeSheet);
    } else {
      sellStockShares_(student, stock, shares, tradeSheet);
    }

  } catch (err) {
    console.error(err);
    writeStockTradeResult_('Trade error: ' + err.message);
  } finally {
    try { lock.releaseLock(); } catch (err) {}
  }
}


function buyStockShares_(student, stock, shares, tradeSheet) {
  const totalCost = round2_(stock.price * shares);

  if (student.balance < totalCost) {
    logStockTrade_(student, 'BUY', stock, shares, stock.price, totalCost, 'Denied', 'Insufficient balance');
    writeStockTradeResult_('Buy denied: ' + student.name + ' needs ' + totalCost + ' but only has ' + student.balance + '.');
    return;
  }

  const newBalance = round2_(student.balance - totalCost);

  updateStudentBalance_(student.row, student.balanceCol, newBalance);
  updatePortfolioBuy_(student, stock, shares, totalCost);
  logStockTrade_(student, 'BUY', stock, shares, stock.price, totalCost, 'Success', '');

  writeStockTradeResult_(student.name + ' bought ' + shares + ' shares of ' + stock.ticker + '. Total: ' + totalCost + '. New balance: ' + newBalance);

  tradeSheet.getRange('B3:B6').clearContent();
}


function sellStockShares_(student, stock, shares, tradeSheet) {
  const holding = findPortfolioHolding_(student.cardId, stock.ticker);

  if (!holding || holding.sharesOwned < shares) {
    logStockTrade_(student, 'SELL', stock, shares, stock.price, 0, 'Denied', 'Not enough shares');
    writeStockTradeResult_('Sell denied: not enough shares of ' + stock.ticker + '.');
    return;
  }

  const saleValue = round2_(stock.price * shares);
  const newBalance = round2_(student.balance + saleValue);

  updateStudentBalance_(student.row, student.balanceCol, newBalance);
  updatePortfolioSell_(holding, shares, stock);
  logStockTrade_(student, 'SELL', stock, shares, stock.price, saleValue, 'Success', '');

  writeStockTradeResult_(student.name + ' sold ' + shares + ' shares of ' + stock.ticker + '. Received: ' + saleValue + '. New balance: ' + newBalance);

  tradeSheet.getRange('B3:B6').clearContent();
}


/*******************************************************
 PORTFOLIO
*******************************************************/

function updatePortfolioBuy_(student, stock, shares, totalCost) {
  const sheet = getOrCreateSheet_(STOCK_CFG.SHEETS.PORTFOLIO);
  ensurePortfolioHeaders_();

  const values = sheet.getDataRange().getValues();
  const headerRowIndex = findHeaderRowByNames_(values, ['Card_ID', 'Ticker']);
  const headers = values[headerRowIndex];

  const cardCol = findHeaderIndex_(headers, ['Card_ID', 'Card ID']);
  const nameCol = findHeaderIndex_(headers, ['Student_Name', 'Student Name']);
  const tickerCol = findHeaderIndex_(headers, ['Ticker']);
  const sharesCol = findHeaderIndex_(headers, ['Shares_Owned', 'Shares Owned', 'Shares']);
  const avgCol = findHeaderIndex_(headers, ['Avg_Buy_Price', 'Avg Buy Price']);
  const costCol = findHeaderIndex_(headers, ['Total_Cost', 'Total Cost']);
  const currentPriceCol = findHeaderIndex_(headers, ['Current Price', 'Current_Price']);
  const marketValueCol = findHeaderIndex_(headers, ['Market Value', 'Market_Value']);
  const unrealizedCol = findHeaderIndex_(headers, ['Unrealized Gain/Loss', 'Unrealized_Gain_Loss', 'Unrealized Gain Loss']);
  const updatedCol = findHeaderIndex_(headers, ['Last_Updated', 'Last Updated']);

  const now = nowString_();
  const targetCard = normalizeCardId_(student.cardId);
  const targetTicker = cleanString_(stock.ticker).toUpperCase();

  for (let i = headerRowIndex + 1; i < values.length; i++) {
    const rowCard = normalizeCardId_(values[i][cardCol]);
    const rowTicker = cleanString_(values[i][tickerCol]).toUpperCase();

    if (rowCard === targetCard && rowTicker === targetTicker) {
      const rowValues = values[i].slice();
      const oldShares = Number(rowValues[sharesCol]) || 0;
      const oldCost = Number(rowValues[costCol]) || 0;

      const newShares = oldShares + shares;
      const newCost = round2_(oldCost + totalCost);
      const avgPrice = round2_(newCost / newShares);
      const currentPrice = Number(stock.price) || 0;
      const marketValue = round2_(newShares * currentPrice);
      const unrealizedGainLoss = round2_((currentPrice - avgPrice) * newShares);

      rowValues[cardCol] = student.cardId;
      rowValues[nameCol] = student.name;
      rowValues[tickerCol] = stock.ticker;
      rowValues[sharesCol] = newShares;
      rowValues[avgCol] = avgPrice;
      rowValues[costCol] = newCost;
      if (currentPriceCol !== -1) rowValues[currentPriceCol] = currentPrice;
      if (marketValueCol !== -1) rowValues[marketValueCol] = marketValue;
      if (unrealizedCol !== -1) rowValues[unrealizedCol] = unrealizedGainLoss;
      if (updatedCol !== -1) rowValues[updatedCol] = now;

      sheet.getRange(i + 1, 1, 1, rowValues.length).setValues([rowValues]);
      return;
    }
  }

  const currentPrice = Number(stock.price) || 0;
  const marketValue = round2_(shares * currentPrice);
  const unrealizedGainLoss = round2_((currentPrice - currentPrice) * shares);

  sheet.getRange(sheet.getLastRow() + 1, 1, 1, STOCK_CFG.HEADERS.PORTFOLIO.length)
    .setValues([[student.cardId, student.name, stock.ticker, shares, stock.price, totalCost, currentPrice, marketValue, unrealizedGainLoss, now]]);
}


function updatePortfolioSell_(holding, sharesToSell, stock) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(STOCK_CFG.SHEETS.PORTFOLIO);
  const now = nowString_();
  const remainingShares = holding.sharesOwned - sharesToSell;

  if (remainingShares <= 0) {
    sheet.deleteRow(holding.row);
    return;
  }

  const currentPrice = stock && stock.price ? Number(stock.price) : holding.avgBuyPrice;
  const remainingCost = round2_(holding.avgBuyPrice * remainingShares);
  const marketValue = round2_(remainingShares * currentPrice);
  const unrealizedGainLoss = round2_((currentPrice - holding.avgBuyPrice) * remainingShares);

  const rowValues = sheet.getRange(holding.row, 1, 1, sheet.getLastColumn()).getValues()[0];

  rowValues[holding.map.shares] = remainingShares;
  rowValues[holding.map.totalCost] = remainingCost;
  if (holding.map.currentPrice !== -1) rowValues[holding.map.currentPrice] = currentPrice;
  if (holding.map.marketValue !== -1) rowValues[holding.map.marketValue] = marketValue;
  if (holding.map.unrealizedGainLoss !== -1) rowValues[holding.map.unrealizedGainLoss] = unrealizedGainLoss;
  if (holding.map.lastUpdated !== -1) rowValues[holding.map.lastUpdated] = now;

  sheet.getRange(holding.row, 1, 1, rowValues.length).setValues([rowValues]);
}


function findPortfolioHolding_(cardId, ticker) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(STOCK_CFG.SHEETS.PORTFOLIO);

  if (!sheet || sheet.getLastRow() < 2) return null;

  ensurePortfolioHeaders_();

  const values = sheet.getDataRange().getValues();
  const headerRowIndex = findHeaderRowByNames_(values, ['Card_ID', 'Ticker']);
  const headers = values[headerRowIndex];

  const cardCol = findHeaderIndex_(headers, ['Card_ID', 'Card ID']);
  const nameCol = findHeaderIndex_(headers, ['Student_Name', 'Student Name']);
  const tickerCol = findHeaderIndex_(headers, ['Ticker']);
  const sharesCol = findHeaderIndex_(headers, ['Shares_Owned', 'Shares Owned', 'Shares']);
  const avgCol = findHeaderIndex_(headers, ['Avg_Buy_Price', 'Avg Buy Price']);
  const costCol = findHeaderIndex_(headers, ['Total_Cost', 'Total Cost']);
  const currentPriceCol = findHeaderIndex_(headers, ['Current Price', 'Current_Price']);
  const marketValueCol = findHeaderIndex_(headers, ['Market Value', 'Market_Value']);
  const unrealizedCol = findHeaderIndex_(headers, ['Unrealized Gain/Loss', 'Unrealized_Gain_Loss', 'Unrealized Gain Loss']);
  const updatedCol = findHeaderIndex_(headers, ['Last_Updated', 'Last Updated']);

  const targetCard = normalizeCardId_(cardId);
  const targetTicker = cleanString_(ticker).toUpperCase();

  for (let i = headerRowIndex + 1; i < values.length; i++) {
    const rowCard = normalizeCardId_(values[i][cardCol]);
    const rowTicker = cleanString_(values[i][tickerCol]).toUpperCase();

    if (rowCard === targetCard && rowTicker === targetTicker) {
      return {
        row: i + 1,
        map: {
          cardId: cardCol,
          studentName: nameCol,
          ticker: tickerCol,
          shares: sharesCol,
          avgBuyPrice: avgCol,
          totalCost: costCol,
          currentPrice: currentPriceCol,
          marketValue: marketValueCol,
          unrealizedGainLoss: unrealizedCol,
          lastUpdated: updatedCol
        },
        cardId: rowCard,
        studentName: nameCol !== -1 ? cleanString_(values[i][nameCol]) : '',
        ticker: rowTicker,
        sharesOwned: Number(values[i][sharesCol]) || 0,
        avgBuyPrice: Number(values[i][avgCol]) || 0,
        totalCost: costCol !== -1 ? Number(values[i][costCol]) || 0 : 0
      };
    }
  }

  return null;
}


function ensurePortfolioHeaders_() {
  const sheet = getOrCreateSheet_(STOCK_CFG.SHEETS.PORTFOLIO);

  if (sheet.getLastRow() < 1 || cleanString_(sheet.getRange(1, 1).getValue()) === '') {
    sheet.getRange(1, 1, 1, STOCK_CFG.HEADERS.PORTFOLIO.length)
      .setValues([STOCK_CFG.HEADERS.PORTFOLIO])
      .setFontWeight('bold');
    return;
  }

  const values = sheet.getDataRange().getValues();
  const headerRowIndex = findHeaderRowByNames_(values, ['Card_ID', 'Ticker']);

  if (headerRowIndex === -1) {
    sheet.clear();
    sheet.getRange(1, 1, 1, STOCK_CFG.HEADERS.PORTFOLIO.length)
      .setValues([STOCK_CFG.HEADERS.PORTFOLIO])
      .setFontWeight('bold');
    return;
  }

  const headers = values[headerRowIndex];
  const currentPriceCol = findHeaderIndex_(headers, ['Current Price', 'Current_Price']);
  const marketValueCol = findHeaderIndex_(headers, ['Market Value', 'Market_Value']);
  const lastUpdatedCol = findHeaderIndex_(headers, ['Last_Updated', 'Last Updated']);

  // If Current Price is missing but Market Value exists, insert Current Price before Market Value.
  if (currentPriceCol === -1 && marketValueCol !== -1) {
    sheet.insertColumnBefore(marketValueCol + 1);
    sheet.getRange(headerRowIndex + 1, marketValueCol + 1).setValue('Current Price');
  }

  // If old 7-column format exists, rewrite/migrate headers by appending the new valuation columns.
  const refreshedHeaders = sheet.getRange(headerRowIndex + 1, 1, 1, sheet.getLastColumn()).getValues()[0];
  const refreshedCurrentPriceCol = findHeaderIndex_(refreshedHeaders, ['Current Price', 'Current_Price']);
  const refreshedMarketValueCol = findHeaderIndex_(refreshedHeaders, ['Market Value', 'Market_Value']);
  const refreshedUnrealizedCol = findHeaderIndex_(refreshedHeaders, ['Unrealized Gain/Loss', 'Unrealized_Gain_Loss', 'Unrealized Gain Loss']);

  if (refreshedCurrentPriceCol === -1 || refreshedMarketValueCol === -1 || refreshedUnrealizedCol === -1) {
    sheet.getRange(headerRowIndex + 1, 1, 1, STOCK_CFG.HEADERS.PORTFOLIO.length)
      .setValues([STOCK_CFG.HEADERS.PORTFOLIO])
      .setFontWeight('bold');
  } else {
    sheet.getRange(headerRowIndex + 1, 1, 1, sheet.getLastColumn()).setFontWeight('bold');
  }
}

  const map = {
    cardId: findHeaderIndex_(headers, ['Card_ID', 'Card ID', 'Card']),
    studentName: findHeaderIndex_(headers, ['Student_Name', 'Student Name', 'Name']),
    ticker: findHeaderIndex_(headers, ['Ticker']),
    sharesOwned: findHeaderIndex_(headers, ['Shares_Owned', 'Shares Owned', 'Shares']),
    avgBuyPrice: findHeaderIndex_(headers, ['Avg_Buy_Price', 'Avg Buy Price', 'Average Buy Price']),
    totalCost: findHeaderIndex_(headers, ['Total_Cost', 'Total Cost']),
    marketValue: findHeaderIndex_(headers, ['Market Value', 'Market_Value']),
    unrealizedGainLoss: findHeaderIndex_(headers, ['Unrealized Gain/Loss', 'Unrealized_Gain_Loss', 'Unrealized Gain Loss']),

  };

  Object.keys(map).forEach(function(key) {
    if (map[key] === -1) {
      throw new Error('Stock_Portfolio missing required column: ' + key);
    }
  });

  return map;
}


function normalizePortfolioRowLength_(rowValues) {
  const row = rowValues.slice();

  while (row.length < STOCK_CFG.HEADERS.PORTFOLIO.length) {
    row.push('');
  }

  return row.slice(0, STOCK_CFG.HEADERS.PORTFOLIO.length);
}


function updateAllPortfolioMarketValues() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const portfolioSheet = ss.getSheetByName(STOCK_CFG.SHEETS.PORTFOLIO);
  const marketSheet = ss.getSheetByName(STOCK_CFG.SHEETS.MARKET);

  if (!portfolioSheet || portfolioSheet.getLastRow() < 2) return;
  if (!marketSheet || marketSheet.getLastRow() < 2) return;

  ensurePortfolioHeaders_();

  const portfolioValues = portfolioSheet.getDataRange().getValues();
  const portfolioHeaderRowIndex = findHeaderRowByNames_(portfolioValues, ['Card_ID', 'Ticker']);
  const portfolioHeaders = portfolioValues[portfolioHeaderRowIndex];

  const tickerCol = findHeaderIndex_(portfolioHeaders, ['Ticker']);
  const sharesCol = findHeaderIndex_(portfolioHeaders, ['Shares_Owned', 'Shares Owned', 'Shares']);
  const avgCol = findHeaderIndex_(portfolioHeaders, ['Avg_Buy_Price', 'Avg Buy Price']);
  const currentPriceCol = findHeaderIndex_(portfolioHeaders, ['Current Price', 'Current_Price']);
  const marketValueCol = findHeaderIndex_(portfolioHeaders, ['Market Value', 'Market_Value']);
  const unrealizedCol = findHeaderIndex_(portfolioHeaders, ['Unrealized Gain/Loss', 'Unrealized_Gain_Loss', 'Unrealized Gain Loss']);
  const updatedCol = findHeaderIndex_(portfolioHeaders, ['Last_Updated', 'Last Updated']);

  const priceMap = getCurrentStockPriceMap_();
  const now = nowString_();

  for (let i = portfolioHeaderRowIndex + 1; i < portfolioValues.length; i++) {
    const ticker = cleanString_(portfolioValues[i][tickerCol]).toUpperCase();
    if (!ticker || !priceMap[ticker]) continue;

    const shares = Number(portfolioValues[i][sharesCol]) || 0;
    const avgBuyPrice = Number(portfolioValues[i][avgCol]) || 0;
    const currentPrice = Number(priceMap[ticker].price) || 0;
    const marketValue = round2_(shares * currentPrice);
    const unrealizedGainLoss = round2_((currentPrice - avgBuyPrice) * shares);

    if (currentPriceCol !== -1) portfolioValues[i][currentPriceCol] = currentPrice;
    if (marketValueCol !== -1) portfolioValues[i][marketValueCol] = marketValue;
    if (unrealizedCol !== -1) portfolioValues[i][unrealizedCol] = unrealizedGainLoss;
    if (updatedCol !== -1) portfolioValues[i][updatedCol] = now;
  }

  portfolioSheet
    .getRange(1, 1, portfolioValues.length, portfolioValues[0].length)
    .setValues(portfolioValues);
}


function updateAllPortfolioMarketValues_() {
  const portfolioSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(STOCK_CFG.SHEETS.PORTFOLIO);
  const marketSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(STOCK_CFG.SHEETS.MARKET);

  if (!portfolioSheet || portfolioSheet.getLastRow() < 2) return;
  if (!marketSheet || marketSheet.getLastRow() < 2) return;

  ensurePortfolioHeaders_();

  const portfolioValues = portfolioSheet.getDataRange().getValues();
  const portfolioHeaders = portfolioValues[0];
  const pMap = getPortfolioHeaderMap_(portfolioHeaders);

  const stockPriceMap = getCurrentStockPriceMap_();
  const now = nowString_();

  for (let i = 1; i < portfolioValues.length; i++) {
    const ticker = cleanString_(portfolioValues[i][pMap.ticker]).toUpperCase();
    const sharesOwned = Number(portfolioValues[i][pMap.sharesOwned]) || 0;
    const avgBuyPrice = Number(portfolioValues[i][pMap.avgBuyPrice]) || 0;

    if (!ticker || sharesOwned <= 0) continue;

    const stock = stockPriceMap[ticker];
    if (!stock) continue;

    const currentPrice = Number(stock.price) || 0;
    const marketValue = round2_(sharesOwned * currentPrice);
    const unrealizedGainLoss = round2_((currentPrice - avgBuyPrice) * sharesOwned);

    portfolioValues[i][pMap.marketValue] = marketValue;
    portfolioValues[i][pMap.unrealizedGainLoss] = unrealizedGainLoss;
    portfolioValues[i][pMap.lastUpdated] = now;
  }

  portfolioSheet
    .getRange(1, 1, portfolioValues.length, STOCK_CFG.HEADERS.PORTFOLIO.length)
    .setValues(portfolioValues.map(normalizePortfolioRowLength_));
}



/*******************************************************
 PRICE / VOLATILITY / MARKET HOURS
*******************************************************/

function updateStockMarketPrices() {
  if (!isMarketOpen_()) {
    if (STOCK_CFG.MARKET_STATUS_TOASTS) {
      SpreadsheetApp.getActiveSpreadsheet().toast(
        'Market closed. Price movements halted until next market open.',
        'Stock Market',
        5
      );
    }
    return;
  }

  const lock = LockService.getScriptLock();

  try {
    lock.waitLock(1000);

    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(STOCK_CFG.SHEETS.MARKET);
    if (!sheet) throw new Error('Stock_Market sheet not found.');

    const lastRow = sheet.getLastRow();
    if (lastRow < 2) return;

    ensureFundamentalRowsForAllMarketInstruments_();
    updateFundamentalScoresFromFinancials_();

    const fundamentalMap = getFundamentalMap_();
    const sectorMap = getSectorMomentumMap_();

    const range = sheet.getRange(2, 1, lastRow - 1, STOCK_CFG.HEADERS.MARKET.length);
    const values = range.getValues();
    const timestamp = nowString_();
    const dateKey = Utilities.formatDate(new Date(), STOCK_CFG.TIMEZONE, 'yyyy-MM-dd');
    const historyRows = [];
    const momentumUpdates = [];

    values.forEach(function(row) {
      const ticker = cleanString_(row[0]).toUpperCase();
      const companyName = cleanString_(row[1]);
      const sector = cleanString_(row[2]);
      const price = Number(row[3]) || 0;
      const assetType = cleanString_(row[11] || 'STOCK').toUpperCase();
      const stableVolatility = Number(row[12]) || 0.5;
      let volatility = Number(row[7]) || 5;
      let trend = cleanString_(row[8]) || 'Stable';
      const active = cleanString_(row[9]).toLowerCase();

      if (!ticker || active !== 'yes' || price <= 0) return;

      if (assetType === 'BOND') {
        volatility = stableVolatility;
        trend = 'Stable';
      }

      const previous = price;
      let move;

      if (assetType === 'BOND') {
        move = randomBetween_(-0.05, 0.05);
      } else {
        const fundamentalBias = getFundamentalPriceBiasPercent_(ticker, sector, trend, fundamentalMap, sectorMap);
        const randomNoise = randomBetween_(
          -(volatility / STOCK_CFG.INTRADAY_VOL_DIVISOR),
          volatility / STOCK_CFG.INTRADAY_VOL_DIVISOR
        );

        move = getIntradayTrendBias_(trend) + fundamentalBias + randomNoise;
      }

      const newPrice = Math.max(1, round2_(previous * (1 + move / 100)));
      const changeAmount = round2_(newPrice - previous);
      const changePercent = round4_((newPrice - previous) / previous);

      row[3] = newPrice;
      row[4] = previous;
      row[5] = changePercent;
      row[6] = changeAmount;
      row[7] = volatility;
      row[8] = trend;
      row[10] = timestamp;

      historyRows.push([timestamp, dateKey, ticker, companyName, newPrice, changePercent]);

      if (assetType !== 'BOND') {
        momentumUpdates.push({ ticker: ticker, momentumDelta: Math.max(-5, Math.min(5, changePercent * 1000)) });
      }
    });

    range.setValues(values);
    appendStockPriceHistoryRows_(historyRows);
    updateAllPortfolioMarketValues();
    updateAllPortfolioMarketValues();
    updateAllPortfolioMarketValues_();
    updateAllPortfolioMarketValues_();
    applyMomentumUpdates_(momentumUpdates);

  } catch (err) {
    console.error(err);
    SpreadsheetApp.getActiveSpreadsheet().toast('Price update error: ' + err.message, 'Stock Market', 5);
  } finally {
    try { lock.releaseLock(); } catch (err) {}
  }
}

function updateCompanyFinancialsFromTrend_() {
  const marketSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(STOCK_CFG.SHEETS.MARKET);
  const financialsSheet = getOrCreateSheet_(STOCK_CFG.SHEETS.FINANCIALS);

  if (!marketSheet) throw new Error('Stock_Market sheet not found.');

  if (financialsSheet.getLastRow() < 1 || cleanString_(financialsSheet.getRange(1, 1).getValue()) === '') {
    setupStockFinancialsSheet_();
  }

  // Make sure newly added instruments, including bonds, have financial rows.
  ensureFinancialRowsForAllMarketInstruments_();

  const marketValues = marketSheet.getRange(2, 1, marketSheet.getLastRow() - 1, STOCK_CFG.HEADERS.MARKET.length).getValues();
  const financialValues = financialsSheet.getRange(1, 1, financialsSheet.getLastRow(), STOCK_CFG.HEADERS.FINANCIALS.length).getValues();

  const headers = financialValues[0];
  const tickerCol = headers.indexOf('Ticker');
  const revenueCol = headers.indexOf('Revenue');
  const profitCol = headers.indexOf('Profit');
  const debtCol = headers.indexOf('Debt');
  const cashCol = headers.indexOf('Cash');
  const growthCol = headers.indexOf('Growth_Rate');
  const riskCol = headers.indexOf('Risk_Rating');
  const notesCol = headers.indexOf('Notes');

  const financialMap = {};

  for (let i = 1; i < financialValues.length; i++) {
    const ticker = cleanString_(financialValues[i][tickerCol]).toUpperCase();
    if (ticker) financialMap[ticker] = i;
  }

  marketValues.forEach(function(row) {
    const ticker = cleanString_(row[0]).toUpperCase();
    const companyName = cleanString_(row[1]);
    const trend = cleanString_(row[8]);
    const assetType = cleanString_(row[11] || 'STOCK').toUpperCase();
    const active = cleanString_(row[9]).toLowerCase();

    if (!ticker || active !== 'yes') return;

    const financialRowIndex = financialMap[ticker];
    if (!financialRowIndex) return;

    const financialRow = financialValues[financialRowIndex];

    let revenue = Number(financialRow[revenueCol]) || randomInt_(5000, 50000);
    let profit = Number(financialRow[profitCol]) || randomInt_(-1000, 10000);
    let debt = Number(financialRow[debtCol]) || randomInt_(0, 25000);
    let cash = Number(financialRow[cashCol]) || randomInt_(500, 15000);
    let growth = Number(financialRow[growthCol]) || 0;

    const modifier = assetType === 'BOND' ? getTrendFinancialModifier_('Stable') : getTrendFinancialModifier_(trend);

    revenue = Math.max(100, Math.round(revenue * (1 + modifier.revenue)));
    profit = Math.round(profit * (1 + modifier.profit));
    debt = Math.max(0, Math.round(debt * (1 + modifier.debt)));
    cash = Math.max(0, Math.round(cash * (1 + modifier.cash)));
    growth = round4_(growth + modifier.growth);

    financialRow[revenueCol] = revenue;
    financialRow[profitCol] = profit;
    financialRow[debtCol] = debt;
    financialRow[cashCol] = cash;
    financialRow[growthCol] = growth;
    financialRow[riskCol] = assignRiskRating_(profit, debt, cash);
    financialRow[notesCol] = companyName + ' updated based on ' + (assetType === 'BOND' ? 'stable bond' : trend) + ' trend.';
  });

  financialsSheet.getRange(1, 1, financialValues.length, financialValues[0].length).setValues(financialValues);
}


function generateMorningNewsFromTrend_() {
  const marketSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(STOCK_CFG.SHEETS.MARKET);
  if (!marketSheet) throw new Error('Stock_Market sheet not found.');

  const values = marketSheet.getRange(2, 1, marketSheet.getLastRow() - 1, STOCK_CFG.HEADERS.MARKET.length).getValues();

  values.forEach(function(row) {
    const ticker = cleanString_(row[0]).toUpperCase();
    const companyName = cleanString_(row[1]);
    const sector = cleanString_(row[2]);
    const trend = cleanString_(row[8]);
    const assetType = cleanString_(row[11] || 'STOCK').toUpperCase();
    const active = cleanString_(row[9]).toLowerCase();

    if (!ticker || active !== 'yes') return;

    const news = assetType === 'BOND'
      ? buildBondNewsEvent_(companyName, sector)
      : buildTrendNewsEvent_(companyName, sector, trend);

    appendMarketNews_(ticker, news.headline, news.body, news.impactType, news.priceImpact, news.volatilityImpact, 'Active');
  });
}


function getTrendFinancialModifier_(trend) {
  trend = cleanString_(trend).toLowerCase();

  if (trend === 'hot') return { revenue: randomBetween_(0.03, 0.08), profit: randomBetween_(0.04, 0.10), debt: randomBetween_(-0.03, 0.02), cash: randomBetween_(0.02, 0.08), growth: randomBetween_(0.01, 0.03) };
  if (trend === 'growth') return { revenue: randomBetween_(0.01, 0.04), profit: randomBetween_(0.01, 0.05), debt: randomBetween_(-0.01, 0.03), cash: randomBetween_(0.00, 0.04), growth: randomBetween_(0.005, 0.02) };
  if (trend === 'decline') return { revenue: randomBetween_(-0.04, -0.01), profit: randomBetween_(-0.07, -0.02), debt: randomBetween_(0.01, 0.05), cash: randomBetween_(-0.05, -0.01), growth: randomBetween_(-0.02, -0.005) };
  if (trend === 'crash') return { revenue: randomBetween_(-0.10, -0.04), profit: randomBetween_(-0.15, -0.05), debt: randomBetween_(0.03, 0.10), cash: randomBetween_(-0.10, -0.03), growth: randomBetween_(-0.05, -0.02) };
  if (trend === 'volatile') return { revenue: randomBetween_(-0.04, 0.05), profit: randomBetween_(-0.08, 0.08), debt: randomBetween_(-0.02, 0.06), cash: randomBetween_(-0.06, 0.05), growth: randomBetween_(-0.025, 0.025) };

  return { revenue: randomBetween_(-0.01, 0.015), profit: randomBetween_(-0.02, 0.025), debt: randomBetween_(-0.01, 0.015), cash: randomBetween_(-0.01, 0.02), growth: randomBetween_(-0.005, 0.008) };
}


function buildTrendNewsEvent_(companyName, sector, trend) {
  trend = cleanString_(trend).toLowerCase();

  if (trend === 'hot') return { headline: companyName + ' surges as demand rises across the ' + sector + ' sector.', body: companyName + ' is seeing strong demand, which may increase investor confidence.', impactType: 'Positive', priceImpact: 0.03, volatilityImpact: 1 };
  if (trend === 'growth') return { headline: companyName + ' reports steady growth and improved market confidence.', body: companyName + ' continues to show positive momentum with moderate risk.', impactType: 'Positive', priceImpact: 0.015, volatilityImpact: 0.5 };
  if (trend === 'decline') return { headline: companyName + ' faces weaker demand and cautious investor sentiment.', body: 'Investors are watching ' + companyName + ' closely as demand weakens.', impactType: 'Negative', priceImpact: -0.015, volatilityImpact: 0.75 };
  if (trend === 'crash') return { headline: companyName + ' drops sharply after major concerns hit the market.', body: companyName + ' is under pressure as investors react to worsening conditions.', impactType: 'Negative', priceImpact: -0.04, volatilityImpact: 2 };
  if (trend === 'volatile') return { headline: companyName + ' sees uncertain trading as investors react to mixed signals.', body: companyName + ' may experience larger price swings as investors disagree on direction.', impactType: 'Mixed', priceImpact: randomBetween_(-0.02, 0.02), volatilityImpact: 1.5 };

  return { headline: companyName + ' remains stable as the ' + sector + ' market shows normal activity.', body: companyName + ' appears steady with no major changes reported.', impactType: 'Neutral', priceImpact: 0, volatilityImpact: 0 };
}


function buildBondNewsEvent_(companyName, sector) {
  return {
    headline: companyName + ' remains steady as investors seek stability.',
    body: companyName + ' is a bond instrument and is less sensitive to market shocks than regular stocks.',
    impactType: 'Neutral',
    priceImpact: 0,
    volatilityImpact: 0
  };
}

function chooseWeeklyShockDay() {
  const props = PropertiesService.getScriptProperties();

  const now = new Date();
  const weekKey = Utilities.formatDate(now, STOCK_CFG.TIMEZONE, 'yyyy-ww');
  const shockDays = getEligibleShockDaysForCurrentWeek_();

  props.setProperty('MARKET_SHOCK_WEEK', weekKey);

  appendMarketNews_('MARKET', 'Weekly Market Outlook', 'The market opens the week with uncertainty. Analysts warn that a possible economic shock may occur later this week, but the exact timing is unknown.', 'Neutral', 0, 0, 'Active');

  SpreadsheetApp.getActiveSpreadsheet().toast('Weekly shock day selected internally.', 'Market Shock', 5);

function checkAndTriggerWeeklyMarketShock() {
  try {
    const props = PropertiesService.getScriptProperties();
    const now = new Date();
    const weekKey = Utilities.formatDate(now, STOCK_CFG.TIMEZONE, 'yyyy-ww');
    const todayName = Utilities.formatDate(now, STOCK_CFG.TIMEZONE, 'EEEE');

    let selectedWeek = props.getProperty('MARKET_SHOCK_WEEK');
    let selectedDay = props.getProperty('MARKET_SHOCK_DAY');
    let alreadyTriggered = props.getProperty('MARKET_SHOCK_TRIGGERED');

    if (selectedWeek !== weekKey || !selectedDay) {
      chooseWeeklyShockDay();
      selectedWeek = props.getProperty('MARKET_SHOCK_WEEK');
      selectedDay = props.getProperty('MARKET_SHOCK_DAY');
      alreadyTriggered = props.getProperty('MARKET_SHOCK_TRIGGERED');
    }

    if (alreadyTriggered === 'YES') return;

    if (todayName !== selectedDay) {
      appendMarketNews_('MARKET', 'Market Stability Report', 'Markets opened without a major economic shock today. Investors remain cautious as uncertainty continues through the week.', 'Neutral', 0, 0, 'Active');
      return;
    }

    const triggered = triggerWeeklyMarketEconomicShock_();

    if (triggered) {
      props.setProperty('MARKET_SHOCK_TRIGGERED', 'YES');
    }

  } catch (err) {
    console.error(err);
    SpreadsheetApp.getActiveSpreadsheet().toast('Weekly shock check error: ' + err.message, 'Market Shock', 5);
  }
}


function triggerWeeklyMarketEconomicShock_() {
  const lock = LockService.getScriptLock();

  try {
    lock.waitLock(1000);

    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(STOCK_CFG.SHEETS.MARKET);
    if (!sheet) throw new Error('Stock_Market sheet not found.');

    const lastRow = sheet.getLastRow();
    if (lastRow < 2) return false;

    const range = sheet.getRange(2, 1, lastRow - 1, STOCK_CFG.HEADERS.MARKET.length);
    const values = range.getValues();

    const activeIndexes = [];

    values.forEach(function(row, index) {
      const ticker = cleanString_(row[0]);
      const active = cleanString_(row[9]).toLowerCase();
      const assetType = cleanString_(row[11] || 'STOCK').toUpperCase();
      const shockProtected = cleanString_(row[13]).toLowerCase();

      if (ticker && active === 'yes' && assetType !== 'BOND' && shockProtected !== 'yes') {
        activeIndexes.push(index);
      }
    });

    shuffleArray_(activeIndexes);

    const affectedCount = Math.ceil(activeIndexes.length * STOCK_CFG.SHOCK_WEEKLY_PERCENT);
    const affectedIndexes = activeIndexes.slice(0, affectedCount);
    const timestamp = nowString_();
    const dateKey = Utilities.formatDate(new Date(), STOCK_CFG.TIMEZONE, 'yyyy-MM-dd');
    const historyRows = [];

    affectedIndexes.forEach(function(index) {
      const row = values[index];

      const oldPrice = Number(row[3]) || 1;
      const dropPercent = randomBetween_(0.02, STOCK_CFG.SHOCK_MAX_DROP_PERCENT);
      const newPrice = Math.max(1, round2_(oldPrice * (1 - dropPercent)));
      const changePercent = round4_((newPrice - oldPrice) / oldPrice);

      row[3] = newPrice;
      row[4] = oldPrice;
      row[5] = changePercent;
      row[6] = round2_(newPrice - oldPrice);
      row[7] = Math.min(15, round2_((Number(row[7]) || 5) + randomBetween_(1, 3)));
      row[8] = Math.random() < 0.55 ? 'Decline' : 'Crash';
      row[10] = timestamp;

      historyRows.push([timestamp, dateKey, cleanString_(row[0]), cleanString_(row[1]), newPrice, changePercent]);
    });

    range.setValues(values);
    appendStockPriceHistoryRows_(historyRows);

    appendMarketNews_(
      'MARKET',
      'Economic Shock Hits the Market',
      'A sudden economic shock hit the classroom market today. Around 60% of non-bond companies were affected, causing sharp price declines and increased volatility across multiple sectors. Bond instruments remained comparatively stable.',
      'Negative',
      -STOCK_CFG.SHOCK_MAX_DROP_PERCENT,
      2,
      'Active'
    );

    SpreadsheetApp.getActiveSpreadsheet().toast('Weekly market shock triggered. ' + affectedCount + ' companies affected.', 'Market Shock', 5);
    return true;

  } catch (err) {
    console.error(err);
    SpreadsheetApp.getActiveSpreadsheet().toast('Weekly market shock error: ' + err.message, 'Market Shock', 5);
    return false;
  } finally {
    try { lock.releaseLock(); } catch (err) {}
  }
}

function applyNewsImpactToStock_(ticker, priceImpact, volatilityImpact) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(STOCK_CFG.SHEETS.MARKET);
  const values = sheet.getRange(1, 1, sheet.getLastRow(), sheet.getLastColumn()).getValues();

  const headers = values[0];

  const tickerCol = headers.indexOf('Ticker');
  const priceCol = headers.indexOf('Current_Price');
  const prevCol = headers.indexOf('Previous_Price');
  const changeCol = headers.indexOf('Change_%');
  const amountCol = headers.indexOf('Change_Amount');
  const volCol = headers.indexOf('Volatility');
  const assetCol = headers.indexOf('Asset_Type');
  const protectedCol = headers.indexOf('Shock_Protected');
  const updatedCol = headers.indexOf('Last_Updated');

  for (let i = 1; i < values.length; i++) {
    if (cleanString_(values[i][tickerCol]).toUpperCase() === ticker) {
      const assetType = assetCol !== -1 ? cleanString_(values[i][assetCol]).toUpperCase() : 'STOCK';
      const shockProtected = protectedCol !== -1 ? cleanString_(values[i][protectedCol]).toLowerCase() : '';

      if (assetType === 'BOND' || shockProtected === 'yes') return;

      const oldPrice = Number(values[i][priceCol]) || 1;
      const newPrice = round2_(oldPrice * (1 + priceImpact));
      const changeAmount = round2_(newPrice - oldPrice);

      sheet.getRange(i + 1, prevCol + 1).setValue(oldPrice);
      sheet.getRange(i + 1, priceCol + 1).setValue(newPrice);
      sheet.getRange(i + 1, changeCol + 1).setValue(priceImpact);
      sheet.getRange(i + 1, amountCol + 1).setValue(changeAmount);
      sheet.getRange(i + 1, volCol + 1).setValue(round2_((Number(values[i][volCol]) || 5) + volatilityImpact));
      sheet.getRange(i + 1, updatedCol + 1).setValue(nowString_());
      logStockPriceHistory_(ticker, cleanString_(values[i][headers.indexOf('Company_Name')]), newPrice, priceImpact);
      updateAllPortfolioMarketValues();
      updateAllPortfolioMarketValues_();
      return;
    }
  }
}

function processStockRating() {
  const lock = LockService.getScriptLock();

  try {
    lock.waitLock(1000);

    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheet = ss.getSheetByName(STOCK_CFG.SHEETS.RATINGS_SUBMIT || 'Stock_Rating_Submit');

    if (!sheet) {
      SpreadsheetApp.getUi().alert('Stock_Rating_Submit sheet not found. Run setupStockRatingSubmitSheet first.');
      return;
    }


    if (!isAnalystSubmissionOpen_()) {
      writeStockRatingSubmitResult_('Rating denied: analyst submissions close at 4:00 PM KST. Submit between 8:00 AM and 4:00 PM KST.');
      return;
    }


    SpreadsheetApp.flush();
    sheet.getRange('B9').activate();
    SpreadsheetApp.flush();
    Utilities.sleep(150);

    writeStockRatingSubmitResult_('Processing rating...');

    const inputs = sheet.getRange('B3:B7').getDisplayValues();

    const cardId = normalizeCardId_(inputs[0][0]);
    const ticker = cleanString_(inputs[1][0]).toUpperCase();
    const rating = cleanString_(inputs[2][0]).toUpperCase();
    const targetPrice = Number(cleanString_(inputs[3][0])) || 0;
    const reason = cleanString_(inputs[4][0]);
    const todayKey = Utilities.formatDate(new Date(), STOCK_CFG.TIMEZONE, 'yyyy-MM-dd');

    if (!cardId || !ticker || !rating || !targetPrice || !reason) {
      writeStockRatingSubmitResult_('Rating failed: Card ID, ticker, rating, target price, and reason are required.');
      return;
    }

    if (!['BUY', 'HOLD', 'SELL'].includes(rating)) {
      writeStockRatingSubmitResult_('Rating failed: Rating must be BUY, HOLD, or SELL.');
      return;
    }

    if (targetPrice <= 0) {
      writeStockRatingSubmitResult_('Rating failed: Target price must be greater than 0.');
      return;
    }

    if (reason.length < 10) {
      writeStockRatingSubmitResult_('Rating failed: Reason must be at least 10 characters.');
      return;
    }

    const student = findStudentByCard_(cardId);

    if (!student) {
      writeStockRatingSubmitResult_('Rating failed: Card ID not found in master Students sheet: ' + cardId);
      return;
    }

    const stock = findStockByTicker_(ticker);

    if (!stock || !stock.active) {
      writeStockRatingSubmitResult_('Rating failed: ticker not found or inactive: ' + ticker);
      return;
    }

    if (analystPredictionAlreadySubmitted_(student.cardId, ticker, todayKey)) {
      writeStockRatingSubmitResult_('Rating denied: ' + student.name + ' already submitted a prediction for ' + ticker + ' today.');
      return;
    }

    addStockRating(student.cardId, student.name, ticker, rating, targetPrice, reason);

    writeStockRatingSubmitResult_(
      student.name + ' submitted a ' + rating + ' rating for ' + ticker +
      ' with target price ' + targetPrice + '. It will be checked after market close.'
    );

    sheet.getRange('B3:B7').clearContent();

  } catch (err) {
    console.error(err);
    writeStockRatingSubmitResult_('Rating error: ' + err.message);
  } finally {
    try {
      lock.releaseLock();
    } catch (err) {}
  }
}


function writeStockRatingSubmitResult_(message) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet()
    .getSheetByName(STOCK_CFG.SHEETS.RATINGS_SUBMIT || 'Stock_Rating_Submit');

  if (sheet) {
    sheet.getRange('B9').setValue(message);
  }
}


function buildTickerValidationRule_() {
  const tickers = getStockTickers_()
    .map(function(stock) {
      return stock.ticker;
    })
    .filter(function(ticker) {
      return ticker;
    });

  if (!tickers.length) return null;

  return SpreadsheetApp.newDataValidation()
    .requireValueInList(tickers, true)
    .setAllowInvalid(false)
    .build();
}




/*******************************************************
 ANALYST PREDICTION REWARD SYSTEM
 Checks end-of-day target price accuracy after market close.

 Reward tiers:
   ±1% = $100
   ±5% = $50
   ±10% = $10

 A prediction can only be rewarded once.
*******************************************************/

function checkAnalystPredictions() {
  const lock = LockService.getScriptLock();

  try {
    lock.waitLock(1000);

    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const ratingsSheet = ss.getSheetByName(STOCK_CFG.SHEETS.RATINGS);

    if (!ratingsSheet || ratingsSheet.getLastRow() < 2) {
      ss.toast('No analyst predictions to check.', 'Stock Ratings', 5);
      return;
    }

    ensureStockRatingsRewardColumns_();

    const values = ratingsSheet.getRange(1, 1, ratingsSheet.getLastRow(), STOCK_CFG.HEADERS.RATINGS.length).getValues();
    const headers = values[0];

    const dateCol = headers.indexOf('Date');
    const cardCol = headers.indexOf('Card_ID');
    const nameCol = headers.indexOf('Student_Name');
    const tickerCol = headers.indexOf('Ticker');
    const targetCol = headers.indexOf('Target_Price');
    const checkDateCol = headers.indexOf('Check_Date');
    const eodPriceCol = headers.indexOf('End_Of_Day_Price');
    const accuracyCol = headers.indexOf('Accuracy_%');
    const statusCol = headers.indexOf('Reward_Status');
    const amountCol = headers.indexOf('Reward_Amount');

    const todayKey = Utilities.formatDate(new Date(), STOCK_CFG.TIMEZONE, 'yyyy-MM-dd');

    const stockMap = getCurrentStockPriceMap_();

    let checkedCount = 0;
    let rewardedCount = 0;
    let totalRewarded = 0;

    for (let i = 1; i < values.length; i++) {
      const row = values[i];

      const predictionDate = valueToDateKeyForStock_(row[dateCol]);
      const rewardStatus = cleanString_(row[statusCol]);
      const ticker = cleanString_(row[tickerCol]).toUpperCase();
      const cardId = normalizeCardId_(row[cardCol]);
      const targetPrice = Number(row[targetCol]) || 0;

      if (!ticker || !targetPrice) continue;

      if (!cardId) {
        row[checkDateCol] = todayKey;
        row[statusCol] = 'Needs Card ID';
        row[amountCol] = 0;
        checkedCount++;
        continue;
      }

      // Only same-day predictions are checked at end of day.
      if (predictionDate !== todayKey) continue;

      // Prevent double rewards.
      if (rewardStatus && rewardStatus !== 'Pending') continue;

      const stock = stockMap[ticker];

      if (!stock) {
        row[checkDateCol] = todayKey;
        row[eodPriceCol] = '';
        row[accuracyCol] = '';
        row[statusCol] = 'No Price Found';
        row[amountCol] = 0;
        checkedCount++;
        continue;
      }

      const eodPrice = stock.price;
      const accuracy = Math.abs(eodPrice - targetPrice) / targetPrice;
      const rewardAmount = getAnalystRewardAmount_(accuracy);

      row[checkDateCol] = todayKey;
      row[eodPriceCol] = eodPrice;
      row[accuracyCol] = round4_(accuracy);
      row[amountCol] = rewardAmount;

      if (rewardAmount > 0) {
        const student = findStudentByCard_(cardId);

        if (student) {
          updateStudentBalance_(student.row, student.balanceCol, Number(student.balance || 0) + rewardAmount);
          row[statusCol] = 'Rewarded';
          rewardedCount++;
          totalRewarded += rewardAmount;
        } else {
          row[statusCol] = 'Student Not Found';
        }
      } else {
        row[statusCol] = 'Missed';
      }

      checkedCount++;
    }

    ratingsSheet
      .getRange(1, 1, values.length, STOCK_CFG.HEADERS.RATINGS.length)
      .setValues(values);

    ss.toast(
      'Checked ' + checkedCount + ' predictions. Rewarded ' + rewardedCount +
      ' students. Total paid: $' + totalRewarded + '.',
      'Stock Ratings',
      8
    );

  } catch (err) {
    console.error(err);
    SpreadsheetApp.getActiveSpreadsheet().toast(
      'Analyst reward check error: ' + err.message,
      'Stock Ratings',
      8
    );
  } finally {
    try {
      lock.releaseLock();
    } catch (err) {}
  }
}


function getAnalystRewardAmount_(accuracyDecimal) {
  // Best reward first:
  // ±1% = $100
  // ±5% = $50
  // ±10% = $10

  if (accuracyDecimal <= STOCK_CFG.ANALYST_REWARD_TIER_3_PERCENT) {
    return STOCK_CFG.ANALYST_REWARD_TIER_3_AMOUNT;
  }

  if (accuracyDecimal <= STOCK_CFG.ANALYST_REWARD_TIER_2_PERCENT) {
    return STOCK_CFG.ANALYST_REWARD_TIER_2_AMOUNT;
  }

  if (accuracyDecimal <= STOCK_CFG.ANALYST_REWARD_TIER_1_PERCENT) {
    return STOCK_CFG.ANALYST_REWARD_TIER_1_AMOUNT;
  }

  return 0;
}


function ensureStockRatingsRewardColumns_() {
  const sheet = getOrCreateSheet_(STOCK_CFG.SHEETS.RATINGS);

  if (sheet.getLastRow() < 1 || cleanString_(sheet.getRange(1, 1).getValue()) === '') {
    setupStockRatingsSheet_();
    return;
  }

  const oldValues = sheet.getRange(1, 1, sheet.getLastRow(), Math.max(sheet.getLastColumn(), STOCK_CFG.HEADERS.RATINGS.length)).getValues();
  const oldHeaders = oldValues[0].map(function(header) {
    return cleanString_(header);
  });

  const alreadyNewFormat = oldHeaders[0] === 'Timestamp' &&
    oldHeaders[1] === 'Date' &&
    oldHeaders[2] === 'Card_ID' &&
    oldHeaders[4] === 'Ticker';

  if (!alreadyNewFormat) {
    migrateOldStockRatingsToRewardFormat_(oldValues, oldHeaders);
    return;
  }

  let changed = false;

  STOCK_CFG.HEADERS.RATINGS.forEach(function(requiredHeader, index) {
    if (oldHeaders[index] !== requiredHeader) {
      sheet.getRange(1, index + 1).setValue(requiredHeader);
      changed = true;
    }
  });

  if (changed) {
    sheet.getRange(1, 1, 1, STOCK_CFG.HEADERS.RATINGS.length).setFontWeight('bold');
  }
}




function migrateOldStockRatingsToRewardFormat_(oldValues, oldHeaders) {
  const sheet = getOrCreateSheet_(STOCK_CFG.SHEETS.RATINGS);

  const timestampCol = oldHeaders.indexOf('Timestamp');
  const oldTickerCol = oldHeaders.indexOf('Ticker');
  const oldStudentNameCol = oldHeaders.indexOf('Student_Name');
  const oldRatingCol = oldHeaders.indexOf('Rating');
  const oldTargetCol = oldHeaders.indexOf('Target_Price');
  const oldReasonCol = oldHeaders.indexOf('Reason');

  const rows = [STOCK_CFG.HEADERS.RATINGS];

  for (let i = 1; i < oldValues.length; i++) {
    const oldRow = oldValues[i];
    const timestamp = timestampCol !== -1 ? oldRow[timestampCol] : '';
    const ticker = oldTickerCol !== -1 ? cleanString_(oldRow[oldTickerCol]).toUpperCase() : '';
    const studentName = oldStudentNameCol !== -1 ? cleanString_(oldRow[oldStudentNameCol]) : '';
    const rating = oldRatingCol !== -1 ? cleanString_(oldRow[oldRatingCol]).toUpperCase() : '';
    const targetPrice = oldTargetCol !== -1 ? Number(oldRow[oldTargetCol]) || '' : '';
    const reason = oldReasonCol !== -1 ? cleanString_(oldRow[oldReasonCol]) : '';

    if (!ticker && !studentName && !rating && !targetPrice && !reason) continue;

    rows.push([
      timestamp || nowString_(),
      valueToDateKeyForStock_(timestamp) || Utilities.formatDate(new Date(), STOCK_CFG.TIMEZONE, 'yyyy-MM-dd'),
      '',
      studentName,
      ticker,
      rating,
      targetPrice,
      reason,
      '',
      '',
      '',
      'Needs Card ID',
      0
    ]);
  }

  sheet.clear();
  sheet.getRange(1, 1, rows.length, STOCK_CFG.HEADERS.RATINGS.length).setValues(rows);
  sheet.getRange(1, 1, 1, STOCK_CFG.HEADERS.RATINGS.length).setFontWeight('bold');
  sheet.setFrozenRows(1);
  sheet.setColumnWidths(1, STOCK_CFG.HEADERS.RATINGS.length, 160);
}


function analystPredictionAlreadySubmitted_(cardId, ticker, dateKey) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(STOCK_CFG.SHEETS.RATINGS);

  if (!sheet || sheet.getLastRow() < 2) return false;

  ensureStockRatingsRewardColumns_();

  const values = sheet.getRange(1, 1, sheet.getLastRow(), STOCK_CFG.HEADERS.RATINGS.length).getValues();
  const headers = values[0];

  const dateCol = headers.indexOf('Date');
  const cardCol = headers.indexOf('Card_ID');
  const tickerCol = headers.indexOf('Ticker');

  if (dateCol === -1 || cardCol === -1 || tickerCol === -1) return false;

  const targetCard = normalizeCardId_(cardId);
  const targetTicker = cleanString_(ticker).toUpperCase();

  for (let i = 1; i < values.length; i++) {
    const rowDate = valueToDateKeyForStock_(values[i][dateCol]);
    const rowCard = normalizeCardId_(values[i][cardCol]);
    const rowTicker = cleanString_(values[i][tickerCol]).toUpperCase();

    if (rowDate === dateKey && rowCard === targetCard && rowTicker === targetTicker) {
      return true;
    }
  }

  return false;
}


function getCurrentStockPriceMap_() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(STOCK_CFG.SHEETS.MARKET);

  if (!sheet || sheet.getLastRow() < 2) return {};

  const values = sheet.getRange(1, 1, sheet.getLastRow(), sheet.getLastColumn()).getValues();
  const headers = values[0];

  const tickerCol = headers.indexOf('Ticker');
  const nameCol = headers.indexOf('Company_Name');
  const priceCol = headers.indexOf('Current_Price');
  const activeCol = headers.indexOf('Active');

  const map = {};

  for (let i = 1; i < values.length; i++) {
    const ticker = cleanString_(values[i][tickerCol]).toUpperCase();
    const active = activeCol !== -1 ? cleanString_(values[i][activeCol]).toLowerCase() : 'yes';

    if (!ticker || active !== 'yes') continue;

    map[ticker] = {
      ticker: ticker,
      name: nameCol !== -1 ? cleanString_(values[i][nameCol]) : ticker,
      price: Number(values[i][priceCol]) || 0
    };
  }

  return map;
}


function valueToDateKeyForStock_(value) {
  if (!value) return '';

  if (Object.prototype.toString.call(value) === '[object Date]' && !isNaN(value)) {
    return Utilities.formatDate(value, STOCK_CFG.TIMEZONE, 'yyyy-MM-dd');
  }

  const text = cleanString_(value);

  if (/^\d{4}-\d{2}-\d{2}$/.test(text)) {
    return text;
  }

  const parsed = new Date(text);

  if (!isNaN(parsed)) {
    return Utilities.formatDate(parsed, STOCK_CFG.TIMEZONE, 'yyyy-MM-dd');
  }

  return text;
}

function updateFundamentalScoresFromFinancials_() {
  ensureFinancialRowsForAllMarketInstruments_();
  ensureFundamentalRowsForAllMarketInstruments_();

  const finSheet = getOrCreateSheet_(STOCK_CFG.SHEETS.FINANCIALS);
  const fundSheet = getOrCreateSheet_(STOCK_CFG.SHEETS.FUNDAMENTALS || 'Stock_Fundamentals');

  if (finSheet.getLastRow() < 2 || fundSheet.getLastRow() < 2) return;

  const financials = finSheet.getRange(1, 1, finSheet.getLastRow(), STOCK_CFG.HEADERS.FINANCIALS.length).getValues();
  const fundamentals = fundSheet.getRange(1, 1, fundSheet.getLastRow(), STOCK_CFG.HEADERS.FUNDAMENTALS.length).getValues();
  const sectorMap = getSectorMomentumMap_();
  const newsMap = getTodayNewsSentimentMap_();

  const finHeaders = financials[0];
  const finTickerCol = finHeaders.indexOf('Ticker');
  const revenueCol = finHeaders.indexOf('Revenue');
  const profitCol = finHeaders.indexOf('Profit');
  const debtCol = finHeaders.indexOf('Debt');
  const cashCol = finHeaders.indexOf('Cash');
  const growthCol = finHeaders.indexOf('Growth_Rate');

  const finMap = {};
  for (let i = 1; i < financials.length; i++) {
    const ticker = cleanString_(financials[i][finTickerCol]).toUpperCase();
    if (ticker) finMap[ticker] = financials[i];
  }

  for (let i = 1; i < fundamentals.length; i++) {
    const row = fundamentals[i];
    const ticker = cleanString_(row[0]).toUpperCase();
    const sector = cleanString_(row[2]);
    const fin = finMap[ticker];
    if (!ticker || !fin) continue;

    const revenue = Number(fin[revenueCol]) || 1;
    const profit = Number(fin[profitCol]) || 0;
    const debt = Number(fin[debtCol]) || 0;
    const cash = Number(fin[cashCol]) || 0;
    const growth = Number(fin[growthCol]) || 0;

    const profitMargin = profit / revenue;
    const debtPressure = debt / Math.max(1, debt + cash + revenue);
    const cashStrength = cash / Math.max(1, revenue);

    let fundamentalStrength = 0;
    fundamentalStrength += clamp_(growth * 250, -40, 40);
    fundamentalStrength += clamp_(profitMargin * 180, -35, 35);
    fundamentalStrength -= clamp_(debtPressure * 120, 0, 35);
    fundamentalStrength += clamp_(cashStrength * 40, 0, 20);

    const newsSentiment = newsMap[ticker] || 0;
    const sectorMomentum = sectorMap[sector] ? Number(sectorMap[sector].momentum) || 0 : 0;
    const oldMomentum = Number(row[8]) || 0;

    const marketSentiment = clamp_(
      fundamentalStrength * 0.45 +
      newsSentiment * 0.25 +
      sectorMomentum * 0.20 +
      oldMomentum * 0.10,
      -100,
      100
    );

    const shockResistance = clamp_(100 - (debtPressure * 120) + (cashStrength * 30), 5, 100);
    const speculationLevel = clamp_(Math.abs(sectorMomentum) + Math.abs(newsSentiment) + randomBetween_(5, 25), 0, 100);

    row[3] = round2_(clamp_(fundamentalStrength, -100, 100));
    row[4] = round2_(marketSentiment);
    row[5] = round2_(newsSentiment);
    row[6] = round2_(shockResistance);
    row[7] = round2_(speculationLevel);
    row[9] = round2_(sectorMomentum);
    row[10] = nowString_();
    row[11] = 'Updated from financials: growth, profit margin, debt pressure, news, sector, and momentum.';
  }

  fundSheet.getRange(1, 1, fundamentals.length, STOCK_CFG.HEADERS.FUNDAMENTALS.length).setValues(fundamentals);
}


function getFundamentalPriceBiasPercent_(ticker, sector, trend, fundamentalMap, sectorMap) {
  const fund = fundamentalMap[ticker] || {};
  const sectorInfo = sectorMap[sector] || { momentum: 0, sensitivity: 1 };

  const fundamentalStrength = Number(fund.fundamentalStrength) || 0;
  const marketSentiment = Number(fund.marketSentiment) || 0;
  const newsSentiment = Number(fund.newsSentiment) || 0;
  const momentum = Number(fund.momentum) || 0;
  const sectorMomentum = Number(sectorInfo.momentum) || 0;
  const macroSentiment = getLatestMacroSentiment_();

  const combinedScore = clamp_(
    fundamentalStrength * 0.30 +
    marketSentiment * 0.25 +
    newsSentiment * 0.15 +
    momentum * 0.10 +
    sectorMomentum * 0.10 +
    macroSentiment * 0.10,
    -100,
    100
  );

  // Convert score to a small per-tick percentage bias.
  // Example: strong +80 score adds about +0.12% per 5-minute tick before randomness.
  return round4_(combinedScore * 0.0015);
}


function updateMacroAndSectorOutlook_() {
  setupMacroOutlookSheet_();
  setupSectorMomentumSheet_();

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const macroSheet = ss.getSheetByName(STOCK_CFG.SHEETS.MACRO || 'Stock_Macro_Outlook');
  const sectorSheet = ss.getSheetByName(STOCK_CFG.SHEETS.SECTOR || 'Stock_Sector_Momentum');

  const gdp = randomInt_(-30, 30);
  const inflation = randomInt_(-30, 30);
  const rates = randomInt_(-30, 30);
  const demand = randomInt_(-30, 30);
  const macroSentiment = clamp_(gdp * 0.30 - inflation * 0.20 - rates * 0.25 + demand * 0.25, -100, 100);
  const todayKey = Utilities.formatDate(new Date(), STOCK_CFG.TIMEZONE, 'yyyy-MM-dd');

  macroSheet.getRange(macroSheet.getLastRow() + 1, 1, 1, STOCK_CFG.HEADERS.MACRO.length).setValues([[
    todayKey,
    gdp,
    inflation,
    rates,
    demand,
    round2_(macroSentiment),
    'Daily macro outlook generated for classroom simulation.'
  ]]);

  if (sectorSheet.getLastRow() >= 2) {
    const values = sectorSheet.getRange(2, 1, sectorSheet.getLastRow() - 1, STOCK_CFG.HEADERS.SECTOR.length).getValues();

    values.forEach(function(row) {
      const sensitivity = Number(row[2]) || 1;
      const oldMomentum = Number(row[1]) || 0;
      const newMomentum = clamp_(oldMomentum * 0.65 + macroSentiment * 0.20 * sensitivity + randomBetween_(-15, 15), -100, 100);
      row[1] = round2_(newMomentum);
      row[3] = nowString_();
      row[4] = 'Updated from macro outlook and random sector rotation.';
    });

    sectorSheet.getRange(2, 1, values.length, STOCK_CFG.HEADERS.SECTOR.length).setValues(values);
  }
}


function generateDailyEarningsReports() {
  const lock = LockService.getScriptLock();

  try {
    lock.waitLock(1000);

    setupEarningsReportsSheet_();
    ensureFinancialRowsForAllMarketInstruments_();
    updateFundamentalScoresFromFinancials_();

    const stocks = getStockTickers_().filter(function(stock) {
      return stock.assetType !== 'BOND';
    });

    if (!stocks.length) return;

    const earningsSheet = getOrCreateSheet_(STOCK_CFG.SHEETS.EARNINGS || 'Stock_Earnings_Reports');
    const count = Math.max(3, Math.ceil(stocks.length * 0.15));
    shuffleArray_(stocks);
    const selected = stocks.slice(0, count);

    const rows = [];

    selected.forEach(function(stock) {
      const surprise = randomBetween_(-1, 1);
      const revenueResult = surprise > 0.25 ? 'Beat' : surprise < -0.25 ? 'Miss' : 'Inline';
      const profitResult = surprise > 0.15 ? 'Beat' : surprise < -0.15 ? 'Miss' : 'Inline';
      const debtResult = surprise > 0.35 ? 'Improved' : surprise < -0.35 ? 'Worse' : 'Stable';
      const guidance = surprise > 0.30 ? 'Raised' : surprise < -0.30 ? 'Lowered' : 'Maintained';
      const priceImpact = clamp_(surprise * 0.06, -0.08, 0.08);

      rows.push([
        nowString_(),
        Utilities.formatDate(new Date(), STOCK_CFG.TIMEZONE, 'yyyy-MM-dd'),
        stock.ticker,
        stock.name,
        revenueResult,
        profitResult,
        debtResult,
        guidance,
        round2_(surprise * 100),
        round4_(priceImpact),
        'Published'
      ]);

      appendMarketNews_(
        stock.ticker,
        stock.name + ' releases earnings report: ' + profitResult + ' profit, guidance ' + guidance + '.',
        'Earnings report: revenue ' + revenueResult + ', profit ' + profitResult + ', debt ' + debtResult + ', guidance ' + guidance + '.',
        priceImpact > 0.01 ? 'Positive' : priceImpact < -0.01 ? 'Negative' : 'Neutral',
        priceImpact,
        Math.abs(priceImpact) * 25,
        'Active'
      );

      applyNewsImpactToStock_(stock.ticker, priceImpact, Math.abs(priceImpact) * 25);
    });

    if (rows.length) {
      earningsSheet.getRange(earningsSheet.getLastRow() + 1, 1, rows.length, STOCK_CFG.HEADERS.EARNINGS.length)
        .setValues(rows);
    }

    updateFundamentalScoresFromFinancials_();

    SpreadsheetApp.getActiveSpreadsheet().toast('Generated ' + rows.length + ' earnings reports.', 'Earnings', 5);

  } catch (err) {
    console.error(err);
    SpreadsheetApp.getActiveSpreadsheet().toast('Earnings report error: ' + err.message, 'Earnings', 5);
  } finally {
    try { lock.releaseLock(); } catch (err) {}
  }
}


function getFundamentalMap_() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(STOCK_CFG.SHEETS.FUNDAMENTALS || 'Stock_Fundamentals');
  const map = {};

  if (!sheet || sheet.getLastRow() < 2) return map;

  const values = sheet.getRange(1, 1, sheet.getLastRow(), STOCK_CFG.HEADERS.FUNDAMENTALS.length).getValues();

  for (let i = 1; i < values.length; i++) {
    const ticker = cleanString_(values[i][0]).toUpperCase();
    if (!ticker) continue;

    map[ticker] = {
      ticker: ticker,
      fundamentalStrength: Number(values[i][3]) || 0,
      marketSentiment: Number(values[i][4]) || 0,
      newsSentiment: Number(values[i][5]) || 0,
      shockResistance: Number(values[i][6]) || 0,
      speculationLevel: Number(values[i][7]) || 0,
      momentum: Number(values[i][8]) || 0,
      sectorMomentum: Number(values[i][9]) || 0
    };
  }

  return map;
}

function getTodayNewsSentimentMap_() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(STOCK_CFG.SHEETS.NEWS);
  const map = {};

  if (!sheet || sheet.getLastRow() < 2) return map;

  const todayKey = Utilities.formatDate(new Date(), STOCK_CFG.TIMEZONE, 'yyyy-MM-dd');
  const values = sheet.getRange(1, 1, sheet.getLastRow(), STOCK_CFG.HEADERS.NEWS.length).getValues();
  const headers = values[0];
  const tsCol = headers.indexOf('Timestamp');
  const tickerCol = headers.indexOf('Ticker');
  const impactCol = headers.indexOf('Impact_Type');
  const priceCol = headers.indexOf('Price_Impact_%');

  for (let i = 1; i < values.length; i++) {
    const ticker = cleanString_(values[i][tickerCol]).toUpperCase();
    if (!ticker || ticker === 'MARKET') continue;

    const dateKey = valueToDateKeyForStock_(values[i][tsCol]);
    if (dateKey !== todayKey) continue;

    const impact = cleanString_(values[i][impactCol]).toLowerCase();
    const priceImpact = Number(values[i][priceCol]) || 0;

    let score = priceImpact * 1000;
    if (impact === 'positive') score += 15;
    if (impact === 'negative') score -= 15;
    if (impact === 'mixed') score += randomBetween_(-10, 10);

    map[ticker] = (map[ticker] || 0) + score;
  }

  Object.keys(map).forEach(function(ticker) {
    map[ticker] = clamp_(map[ticker], -100, 100);
  });

  return map;
}


function applyMomentumUpdates_(updates) {
  if (!updates || !updates.length) return;

  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(STOCK_CFG.SHEETS.FUNDAMENTALS || 'Stock_Fundamentals');
  if (!sheet || sheet.getLastRow() < 2) return;

  const values = sheet.getRange(1, 1, sheet.getLastRow(), STOCK_CFG.HEADERS.FUNDAMENTALS.length).getValues();
  const updateMap = {};

  updates.forEach(function(update) {
    updateMap[update.ticker] = (updateMap[update.ticker] || 0) + update.momentumDelta;
  });

  for (let i = 1; i < values.length; i++) {
    const ticker = cleanString_(values[i][0]).toUpperCase();
    if (!updateMap[ticker]) continue;

    const oldMomentum = Number(values[i][8]) || 0;
    values[i][8] = round2_(clamp_(oldMomentum * 0.92 + updateMap[ticker], -100, 100));
    values[i][10] = nowString_();
  }

  sheet.getRange(1, 1, values.length, STOCK_CFG.HEADERS.FUNDAMENTALS.length).setValues(values);
}


function clamp_(value, min, max) {
  value = Number(value) || 0;
  return Math.max(min, Math.min(max, value));
}

function findStudentByCard_(cardId) {
  const masterSS = SpreadsheetApp.openById(MASTER_SPREADSHEET_ID);
  const sheet = masterSS.getSheetByName(STOCK_CFG.SHEETS.STUDENTS);

  if (!sheet) throw new Error('Master Students sheet not found.');

  const HEADER_ROW = 3;

  const values = sheet.getRange(HEADER_ROW, 1, sheet.getLastRow() - HEADER_ROW + 1, sheet.getLastColumn()).getValues();

  const headers = values[0].map(function(header) {
    return normalizeHeader_(header);
  });

  const cardCol = findHeaderIndexFlexible_(headers, ['card_id', 'card id', 'cardid', 'card']);
  const nameCol = findHeaderIndexFlexible_(headers, ['student_name', 'student name', 'name']);
  const balanceCol = findHeaderIndexFlexible_(headers, ['balance', 'current_balance', 'current balance']);
  const activeCol = findHeaderIndexFlexible_(headers, ['active', 'status']);

  if (cardCol === -1) throw new Error('Students sheet missing Card_ID on row 3.');
  if (nameCol === -1) throw new Error('Students sheet missing Student_Name on row 3.');
  if (balanceCol === -1) throw new Error('Students sheet missing Balance on row 3.');

  const targetCard = normalizeCardId_(cardId);

  for (let i = 1; i < values.length; i++) {
    const rowCard = normalizeCardId_(values[i][cardCol]);

    if (rowCard === targetCard) {
      const active = activeCol !== -1 ? cleanString_(values[i][activeCol]).toLowerCase() : 'yes';

      if (active && !['yes', 'y', 'true', 'active', '1'].includes(active)) return null;

      return {
        row: HEADER_ROW + i,
        cardId: rowCard,
        name: cleanString_(values[i][nameCol]),
        balance: Number(values[i][balanceCol]) || 0,
        balanceCol: balanceCol + 1
      };
    }
  }

  return null;
}


function updateStudentBalance_(row, balanceCol, newBalance) {

function findStockByTicker_(ticker) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(STOCK_CFG.SHEETS.MARKET);

  if (!sheet) throw new Error('Stock_Market sheet not found.');

  const values = sheet.getRange(1, 1, sheet.getLastRow(), sheet.getLastColumn()).getValues();

  const tickerCol = headers.indexOf('Ticker');
  const nameCol = headers.indexOf('Company_Name');
  const priceCol = headers.indexOf('Current_Price');
  const activeCol = headers.indexOf('Active');
  const assetCol = headers.indexOf('Asset_Type');

  for (let i = 1; i < values.length; i++) {

    if (rowTicker === ticker) {
      return {
        row: i + 1,
        ticker: rowTicker,
        name: cleanString_(values[i][nameCol]),
        price: Number(values[i][priceCol]) || 0,
        active: cleanString_(values[i][activeCol]).toLowerCase() === 'yes',
        assetType: assetCol !== -1 ? cleanString_(values[i][assetCol]).toUpperCase() : 'STOCK'
      };
    }
  }

  return null;
}

function getIntradayTrendBias_(trend) {
  trend = cleanString_(trend).toLowerCase();

  if (trend === 'hot') return 0.18;
  if (trend === 'growth') return 0.08;
  if (trend === 'decline') return -0.08;
  if (trend === 'crash') return -0.18;

  return 0;
}


function updateDailyVolatility_(volatility, trend) {
  trend = cleanString_(trend).toLowerCase();

  let change = randomBetween_(-1, 1);

  if (trend === 'hot' || trend === 'crash' || trend === 'volatile') {
    change += randomBetween_(0, 1.25);
  }

  if (trend === 'stable') {
    change -= randomBetween_(0, 0.75);
  }

  return round2_(Math.max(2, Math.min(15, volatility + change)));
}


function updateDailyTrend_() {
  const roll = Math.random();

  if (roll < 0.08) return 'Hot';
  if (roll < 0.28) return 'Growth';
  if (roll < 0.68) return 'Stable';
  if (roll < 0.90) return 'Decline';
  if (roll < 0.97) return 'Volatile';

  return 'Crash';
}


function assignRiskRating_(profit, debt, cash) {
  if (profit < 0 && debt > cash) return 'High';
  if (debt > cash * 2) return 'Medium-High';
  if (profit > 5000 && cash > debt) return 'Low';
  return 'Medium';
}



function getEligibleShockDaysForCurrentWeek_() {
  const todayName = Utilities.formatDate(new Date(), STOCK_CFG.TIMEZONE, 'EEEE');
  const weekdays = ['Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday'];
  const shockDays = ['Tuesday', 'Wednesday', 'Thursday', 'Friday'];
  const todayIndex = weekdays.indexOf(todayName);

  // Normal Monday selection keeps the event unpredictable later in the week.
  if (todayName === 'Monday') return shockDays;

  // If the script is installed midweek or the Monday selector was missed,
  // choose from today or later so the shock cannot be assigned to a past day.
  if (todayIndex >= 1 && todayIndex <= 4) {
    return weekdays.slice(todayIndex, 5);
  }

  // Weekend fallback: select from the next trading week.
  return shockDays;
}

function findHeaderRowByNames_(values, requiredHeaders) {
  for (let r = 0; r < Math.min(values.length, 10); r++) {
    const headers = values[r].map(function(header) {
      return normalizeHeader_(header);
    });

    const foundAll = requiredHeaders.every(function(requiredHeader) {
      return headers.indexOf(normalizeHeader_(requiredHeader)) !== -1;
    });

    if (foundAll) return r;
  }

  return -1;
}


function findHeaderIndex_(headers, possibleNames) {
  const normalizedHeaders = headers.map(function(header) {
    return normalizeHeader_(header);
  });

  for (let i = 0; i < possibleNames.length; i++) {
    const target = normalizeHeader_(possibleNames[i]);
    const index = normalizedHeaders.indexOf(target);
    if (index !== -1) return index;
  }

  return -1;
}

function nowString_() {
  return Utilities.formatDate(new Date(), STOCK_CFG.TIMEZONE, 'yyyy-MM-dd HH:mm:ss');
}


function randomBetween_(min, max) {
  return Math.random() * (max - min) + min;
}


function randomInt_(min, max) {
  return Math.floor(Math.random() * (max - min + 1)) + min;
}


function round2_(num) {
  return Math.round(Number(num) * 100) / 100;
}


function round4_(num) {
  return Math.round(Number(num) * 10000) / 10000;
}


function normalizeCardId_(value) {
  return cleanString_(value)
    .toLowerCase()
    .replace(/[\u200B-\u200D\uFEFF]/g, '')
    .replace(/\s+/g, '')
    .replace(/[^a-z0-9]/g, '');
}


function normalizeHeader_(value) {
  return cleanString_(value)
    .toLowerCase()
    .replace(/[\u200B-\u200D\uFEFF]/g, '')
    .replace(/\s+/g, '_')
    .replace(/-/g, '_');
}


function findHeaderIndexFlexible_(normalizedHeaders, possibleNames) {
  const normalizedPossible = possibleNames.map(function(name) {
    return normalizeHeader_(name);
  });

  for (let i = 0; i < normalizedPossible.length; i++) {
    const index = normalizedHeaders.indexOf(normalizedPossible[i]);

    if (index !== -1) return index;
  }

  return -1;
}


function shuffleArray_(array) {
  for (let i = array.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    const temp = array[i];
    array[i] = array[j];
    array[j] = temp;
  }
}


function cleanString_(value) {
  if (value === null || typeof value === 'undefined') return '';

  return String(value)
    .replace(/[\u200B-\u200D\uFEFF]/g, '')
    .trim();
}
