/*******************************************************
 DAILY ATTENDANCE TODAY-ONLY CLEANUP
 Keeps Daily_Attendance as a live current-day sheet.
 Attendance_Log remains the historical record.
*******************************************************/

function purgeOldDailyAttendanceRows_(todayKey) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(CFG.SHEETS.DAILY_ATTENDANCE);

  // Require at least header + data area
  if (!sheet || sheet.getLastRow() < 4) return;

  const values = getUsedValues_(sheet);

  if (!values || values.length < 2) return;

  const headerRowIndex = findHeaderRowByNames_(values, ['Date', 'Card_ID', 'Status']);

  if (headerRowIndex === -1) return;

  const headers = values[headerRowIndex];

  const dateCol = findHeaderIndex_(headers, ['Date']);

  if (dateCol === -1) return;

  // Convert zero-based array index → actual sheet row
  const headerSheetRow = headerRowIndex + 1;

  // Data should start BELOW header row
  const firstDataRow = headerSheetRow + 1;

  const rowsToKeep = [];

  // Keep only today's attendance rows
  for (let i = headerRowIndex + 1; i < values.length; i++) {
    const rowDate = valueToDateKey_(values[i][dateCol]);

    if (rowDate === todayKey) {
      rowsToKeep.push(values[i]);
    }
  }

  const lastRow = sheet.getLastRow();
  const lastCol = sheet.getLastColumn();

  // Clear old attendance rows ONLY below headers
  if (lastRow >= firstDataRow) {
    sheet
      .getRange(
        firstDataRow,
        1,
        lastRow - firstDataRow + 1,
        lastCol
      )
      .clearContent();
  }

  // Re-write today's rows starting at row 4 (if headers on row 3)
  if (rowsToKeep.length > 0) {
    sheet
      .getRange(
        firstDataRow,
        1,
        rowsToKeep.length,
        headers.length
      )
      .setValues(rowsToKeep);
  }

  clearOldAttendanceIndexProperties_(todayKey);
}


function clearOldAttendanceIndexProperties_(todayKey) {
  const props = PropertiesService.getDocumentProperties();
  const allProps = props.getProperties();

  Object.keys(allProps).forEach(function(key) {
    if (
      key.indexOf('ATT_IDX_') === 0 &&
      key.indexOf('ATT_IDX_' + todayKey + '_') !== 0
    ) {
      props.deleteProperty(key);
    }
  });
}

function processStoreKioskPurchase() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(CFG.SHEETS.STORE_KIOSK || 'Store_Kiosk');

  if (!sheet) {
    SpreadsheetApp.getUi().alert('Store_Kiosk sheet not found. Run setupStoreKioskTab first.');
    return;
  }

  const lock = LockService.getScriptLock();

  if (!lock.tryLock(3000)) {
    setStatus_(sheet, 'Store kiosk is busy. Try again.');
    return;
  }

  try {
    /*******************************************************
     STORE KIOSK ACTIVE-CELL COMMIT PATCH

     WHY:
     Google Sheets sometimes does not fully commit the value in an
     actively selected/edited input cell before a drawing/button click
     runs the assigned script. This caused purchases to process as if
     B4/B6/B7 were still blank or using the previous value.

     WHAT THIS DOES:
     1. Flushes pending edits.
     2. Moves selection to the status cell B9 to force the active input
        cell to commit.
     3. Flushes again and waits briefly before reading inputs.
     4. Reads getDisplayValues() so the script uses the visible kiosk text.
    *******************************************************/
    SpreadsheetApp.flush();
    sheet.getRange('B9').activate();
    SpreadsheetApp.flush();
    Utilities.sleep(150);

    clearStatus_(sheet);
    setStatus_(sheet, 'Processing store purchase...');

    const inputValues = sheet.getRange('B4:B7').getDisplayValues();
    const cardId = cleanString_(inputValues[0][0]);
    const itemId = cleanString_(inputValues[2][0]);
    const quantityRaw = cleanString_(inputValues[3][0]);
    const quantity = Number(quantityRaw);

    if (!cardId || !itemId || !quantityRaw) {
      setStatus_(sheet, 'Purchase failed: Student Card ID, Store Item ID, and Quantity are all required.');
      return;
    }

    if (!Number.isInteger(quantity) || quantity <= 0) {
      setStatus_(sheet, 'Purchase failed: Quantity must be a whole number greater than 0.');
      return;
    }

    const student = findStudentByCard_(cardId);

    if (!student) {
      setStatus_(sheet, 'Unknown Card ID.');
      clearStoreKioskInputs_(sheet, true);
      return;
    }

    const activeValue = cleanString_(student.active);
    if (activeValue && !isTruthy_(activeValue)) {
      setStatus_(sheet, 'Denied: ' + student.name + ' is inactive.');
      clearStoreKioskInputs_(sheet, true);
      return;
    }

    const item = findStoreItemById_(itemId);

    if (!item) {
      setStatus_(sheet, 'Store item not found: ' + itemId);
      sheet.getRange('B6').clearContent();
      return;
    }

    if (!isTruthy_(item.active)) {
      setStatus_(sheet, 'Denied: ' + item.name + ' is not active.');
      sheet.getRange('B6').clearContent();
      return;
    }

    const currentInventory = item.inventory !== '' && !isNaN(Number(item.inventory))
      ? Number(item.inventory)
      : null;

    if (currentInventory !== null && currentInventory <= 0) {
      setStatus_(sheet, 'Denied: ' + item.name + ' is out of stock.');
      sheet.getRange('B6:B7').clearContent();
      return;
    }

    if (currentInventory !== null && quantity > currentInventory) {
      setStatus_(
        sheet,
        'Purchase failed: only ' + currentInventory + ' available for ' + item.name + '.'
      );
      return;
    }

    const totalCost = Number(item.price) * quantity;

    const result = applyBalanceChange_(
      student,
      'STORE_PURCHASE',
      totalCost,
      'Self-service store purchase: ' + item.name + ' x' + quantity,
      item.id,
      item.name
    );

    if (!result.success) {
      setStatus_(sheet, result.message);
      clearStoreKioskInputs_(sheet, true);
      return;
    }

    reduceStoreInventoryByQuantity_(item, quantity);
    updateStudentInventory_(student, item, quantity);

    setStatus_(
      sheet,
      student.name + ' bought ' + quantity + ' x ' + item.name +
      ' for ' + totalCost + '. New balance: ' + result.endingBalance
    );

    clearStoreKioskInputs_(sheet, true);
  } catch (err) {
    console.error(err);
    setStatus_(sheet, 'Purchase error: ' + err.message);
  } finally {
    lock.releaseLock();
  }
}


function clearStoreKioskInputs_(sheet, keepStatus) {
  sheet.getRange('B4').clearContent();
  sheet.getRange('B6').clearContent();
  sheet.getRange('B7').clearContent();

  if (!keepStatus) {
    setStatus_(
      sheet,
      'Scan card, choose an item, enter quantity, then press Purchase.'
    );
  }
}

/*******************************************************
 ON EDIT
*******************************************************/

function onEdit(e) {
  if (!e || !e.range) return;

  const sheet = e.range.getSheet();
  const sheetName = sheet.getName();
  const cell = e.range.getA1Notation();
  const row = e.range.getRow();

  // Ignore header-row edits and only run exact scan-cell routes below.
  if (row === 1) return;

  try {
    if (sheetName === CFG.SHEETS.SCAN && cell === CFG.CELLS.MAIN_SCAN) {
      clearStatus_(sheet);
      setStatus_(sheet, 'Processing scan...');
      processMainScannerEdit_(e);
      return;
    }

    if (sheetName === CFG.SHEETS.STUDENT_PROFILE && cell === CFG.CELLS.PROFILE_SCAN) {
      clearStatus_(sheet);
      setStatus_(sheet, 'Loading student profile...');
      processStudentProfileScan_(e);
      return;
    }
  } catch (err) {
    setStatus_(sheet, 'Script error: ' + err.message);
    console.error(err);
  }
}


/*******************************************************
 MAIN SCANNER LOGIC
*******************************************************/

function processMainScannerEdit_(e) {
  const lock = LockService.getScriptLock();

  if (!lock.tryLock(250)) {
    writeMainResult_('Scanner is busy. Try again.');
    return;
  }

  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const scanSheet = ss.getSheetByName(CFG.SHEETS.SCAN);

    const rawCardId = e.value;
    const cardId = cleanString_(rawCardId);

    if (!cardId) {
      finishMainScan_('No Card ID detected. Make sure the scanner is typing into Scan!B3.', scanSheet);
      return;
    }

    // One read for all scan controls instead of several getRange calls.
    const controls = scanSheet.getRange('B4:B8').getValues();
    const scanContext = {
      mode: cleanString_(controls[0][0]).toUpperCase(),
      amount: Number(controls[1][0]) || 0,
      classPeriod: cleanString_(controls[2][0]),
      itemId: cleanString_(controls[3][0]),
      note: cleanString_(controls[4][0])
    };

    if (CFG.MODES.indexOf(scanContext.mode) === -1) {
      finishMainScan_('Error: Choose a valid mode in B4.', scanSheet);
      return;
    }

    const student = findStudentByCard_(cardId);

    if (!student) {
      logUnknownCard_(cardId, scanContext.mode, scanContext.note);
      finishMainScan_('Unknown card: ' + cardId + '. Add this Card_ID to the Students tab.', scanSheet);
      return;
    }

    // Blank Active cells are treated as active. Only explicit inactive/no values block the scan.
    const activeValue = cleanString_(student.active);
    if (activeValue && !isTruthy_(activeValue)) {
      finishMainScan_('Denied: ' + student.name + ' is marked inactive.', scanSheet);
      return;
    }

    if (scanContext.mode === 'ATTENDANCE') {
      processAttendance_(student, scanContext, scanSheet);
      return;
    }

    if (scanContext.mode === 'STORE_PURCHASE') {
      processStorePurchase_(student, scanContext, scanSheet);
      return;
    }

    processMoneyMode_(student, scanContext.mode, scanContext, scanSheet);
  } finally {
    lock.releaseLock();
  }
}




function processMoneyMode_(student, mode, scanContext, scanSheet) {
  const amount = Number(scanContext && scanContext.amount) || 0;
  const note = scanContext ? cleanString_(scanContext.note) : '';

  if (!amount || amount <= 0) {
    finishMainScan_('Error: Enter a valid amount in B5.', scanSheet);
    return;
  }

  const result = applyBalanceChange_(student, mode, amount, note, '', '');

  if (!result.success) {
    finishMainScan_(result.message, scanSheet);
    return;
  }

  finishMainScan_(student.name + ' ' + result.symbol + amount + '. New balance: ' + result.endingBalance, scanSheet);
}




function processStorePurchase_(student, scanContext, scanSheet) {
  const itemId = scanContext ? cleanString_(scanContext.itemId) : '';

  if (!itemId) {
    finishMainScan_('Error: Enter a Store Item ID in B7 before scanning.', scanSheet);
    return;
  }

  const item = findStoreItemById_(itemId);

  if (!item) {
    finishMainScan_('Error: Store item not found: ' + itemId, scanSheet);
    return;
  }

  if (!isTruthy_(item.active)) {
    finishMainScan_('Denied: ' + item.name + ' is not currently active.', scanSheet);
    return;
  }

  if (item.inventory !== '' && !isNaN(Number(item.inventory)) && Number(item.inventory) <= 0) {
    finishMainScan_('Denied: ' + item.name + ' is out of inventory.', scanSheet);
    return;
  }

  const result = applyBalanceChange_(
    student,
    'STORE_PURCHASE',
    Number(item.price),
    'Store purchase: ' + item.name,
    item.id,
    item.name
  );

  if (!result.success) {
    finishMainScan_(result.message, scanSheet);
    return;
  }

  reduceStoreInventory_(item);
  updateStudentInventory_(student, item, 1);

  finishMainScan_(
    student.name + ' bought ' + item.name + ' for ' + item.price + '. New balance: ' + result.endingBalance,
    scanSheet
  );
}




function processAttendance_(student, scanContext, scanSheet) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const attendanceLogSheet = ss.getSheetByName(CFG.SHEETS.ATTENDANCE_LOG);
  const dailySheet = ss.getSheetByName(CFG.SHEETS.DAILY_ATTENDANCE);

  const now = new Date();
  const timestampKST = formatDate_(now, 'yyyy-MM-dd HH:mm:ss');
  const dateKey = formatDate_(now, 'yyyy-MM-dd');
  const timeKey = formatDate_(now, 'HH:mm:ss');

  // Keeps Daily_Attendance as TODAY ONLY.
  purgeOldDailyAttendanceRows_(dateKey);

  let classPeriod = scanContext ? cleanString_(scanContext.classPeriod) : '';
  if (!classPeriod) classPeriod = getSettingWithAliases_(['Default_Class_Period', 'Default_Class'], 'Homeroom');

  // Fast duplicate check: no full Daily_Attendance read during every scan.
  if (attendanceAlreadyRecorded_(student.cardId, dateKey, classPeriod)) {
    appendRowsToSheet_(attendanceLogSheet, [[
      timestampKST,
      dateKey,
      student.cardId,
      student.name,
      student.grade,
      student.homeroom,
      classPeriod,
      'Duplicate',
      timeKey,
      '',
      0,
      'Duplicate scan',
      'Denied - duplicate attendance'
    ]]);

    finishMainScan_(student.name + ' is already marked for ' + classPeriod + ' today.', scanSheet);
    return;
  }

  const status = getAttendanceStatus_(now);

  if (status === 'Too Early') {
    appendRowsToSheet_(attendanceLogSheet, [[
      timestampKST,
      dateKey,
      student.cardId,
      student.name,
      student.grade,
      student.homeroom,
      classPeriod,
      'Too Early',
      timeKey,
      '',
      0,
      'Attempted check-in before 07:40',
      'Denied - before attendance window'
    ]]);

    finishMainScan_(
      student.name + ' cannot check in before 07:40. Attendance opens at 07:40.',
      scanSheet
    );

    return;
  }

  // Attendance streaks count on-time attendance only.
  // Present = continue/increase streak. Late = reset streak to 0.
  const newStreak = status === 'Present'
    ? updateAttendanceStreak_(student, dateKey)
    : resetAttendanceStreak_(student, dateKey);

  const onTimeReward = Number(getSettingWithAliases_(['On_Time_Attendance_Reward', 'On_Time_Bonus'], 0)) || 0;
  const latePenaltySetting = Number(getSettingWithAliases_(['Late_Penalty', 'Late_Bonus'], 0)) || 0;
  const latePenalty = Math.abs(latePenaltySetting);

  let bonusApplied = 0;
  let note = '';

  if (status === 'Present' && onTimeReward > 0) {
    const rewardResult = applyBalanceChange_(
      student,
      'REWARD',
      onTimeReward,
      'On-time attendance reward',
      '',
      ''
    );

    if (rewardResult.success) {
      bonusApplied += onTimeReward;
      note = 'On-time reward applied';
      student.balance = rewardResult.endingBalance;
    }
  }

  if (status === 'Late' && latePenalty > 0) {
    const penaltyResult = applyBalanceChange_(
      student,
      'FINE',
      latePenalty,
      'Late attendance penalty',
      '',
      ''
    );

    if (penaltyResult.success) {
      note = 'Late penalty applied';
      student.balance = penaltyResult.endingBalance;
    } else {
      note = 'Late penalty not applied: insufficient balance';
    }
  }

  // Streak bonus only applies to on-time attendance.
  const streakBonus = status === 'Present'
    ? maybeApplyStreakBonus_(student.cardId)
    : { amount: 0, message: '' };

  if (streakBonus.amount > 0) {
    bonusApplied += streakBonus.amount;
    note = note ? note + '; ' + streakBonus.message : streakBonus.message;
  }

  const attendanceLogRow = [
    timestampKST,
    dateKey,
    student.cardId,
    student.name,
    student.grade,
    student.homeroom,
    classPeriod,
    status,
    timeKey,
    newStreak,
    bonusApplied,
    note,
    'Success'
  ];

  const dailyAttendanceRow = [
    dateKey,
    student.cardId,
    student.name,
    student.grade,
    student.homeroom,
    classPeriod,
    status,
    timeKey,
    newStreak,
    bonusApplied,
    note
  ];

  appendRowsToSheet_(attendanceLogSheet, [attendanceLogRow]);
  appendRowsToSheet_(dailySheet, [dailyAttendanceRow]);
  markAttendanceRecorded_(student.cardId, dateKey, classPeriod);

  let resultMessage = student.name + ' marked ' + status + ' at ' + timeKey + ' KST.';

  if (bonusApplied > 0) {
    resultMessage += ' Bonus: +' + bonusApplied + '.';
  }

  finishMainScan_(resultMessage, scanSheet);
}




function attendanceAlreadyRecorded_(cardId, dateKey, classPeriod) {
  const key = buildAttendanceIndexKey_(cardId, dateKey, classPeriod);
  const cache = CacheService.getDocumentCache();

  if (cache && cache.get(key) === '1') return true;

  const props = PropertiesService.getDocumentProperties();
  const value = props.getProperty(key);

  if (value === '1') {
    if (cache) cache.put(key, '1', 21600);
    return true;
  }

  return false;
}




/*******************************************************
 STUDENT PROFILE SCAN
*******************************************************/

function processStudentProfileScan_(e) {
  const lock = LockService.getScriptLock();
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(CFG.SHEETS.STUDENT_PROFILE);

  if (!lock.tryLock(500)) {
    setStatus_(sheet, 'Profile scanner is busy. Try again.');
    return;
  }

  try {
    const scanCell = sheet.getRange(CFG.CELLS.PROFILE_SCAN);
    const rawInput = cleanString_(e.value);
    const now = new Date();
    const nowMs = now.getTime();

    const lockUntilMs = Number(getProfileSecurityProperty_('lockUntilMs', 0)) || 0;

    if (lockUntilMs && nowMs < lockUntilMs) {
      const remainingSeconds = Math.ceil((lockUntilMs - nowMs) / 1000);
      const remainingMinutes = Math.ceil(remainingSeconds / 60);

      scanCell.clearContent();
      setStatus_(
        sheet,
        'Locked. Too many attempts. Try again in about ' + remainingMinutes + ' minute(s).'
      );

      return;
    }

    if (!rawInput) {
      setStatus_(sheet, 'Scan a card in B4 to load a student profile.');
      return;
    }

    if (rawInput.length > PROFILE_SECURITY.MAX_CARD_LENGTH) {
      registerProfileScanAttempt_();
      scanCell.clearContent();
      setStatus_(
        sheet,
        'Invalid Card ID. Maximum ' + PROFILE_SECURITY.MAX_CARD_LENGTH + ' characters allowed.'
      );
      checkAndApplyProfileLock_(sheet);
      return;
    }

    registerProfileScanAttempt_();

    if (checkAndApplyProfileLock_(sheet)) {
      scanCell.clearContent();
      return;
    }

    const student = findStudentByCard_(rawInput);
    const time = formatDate_(new Date(), 'HH:mm:ss');

    if (!student) {
      setStatus_(sheet, 'Unknown card: ' + rawInput);
      return;
    }

    setStatus_(sheet, 'Loaded ' + student.name + ' at ' + time + ' KST');
  } finally {
    lock.releaseLock();
  }
}


/*******************************************************
 BALANCE / TRANSACTION HELPERS
*******************************************************/

function applyBalanceChange_(student, mode, amount, note, itemId, itemName) {
  amount = Number(amount);

  if (!student) {
    return {
      success: false,
      message: 'Error: Student not found.'
    };
  }

  if (!amount || amount <= 0) {
    return {
      success: false,
      message: 'Error: Invalid amount.'
    };
  }

  const creditModes = ['DEPOSIT', 'REWARD', 'PAYROLL'];
  const debitModes = ['WITHDRAW', 'FINE', 'STORE_PURCHASE'];

  const isCredit = creditModes.indexOf(mode) !== -1;
  const isDebit = debitModes.indexOf(mode) !== -1;

  if (!isCredit && !isDebit) {
    return {
      success: false,
      message: 'Error: Unknown money mode: ' + mode
    };
  }

  const startingBalance = Number(student.balance) || 0;
  const endingBalance = isCredit ? startingBalance + amount : startingBalance - amount;
  const allowNegative = isTruthy_(getSettingWithAliases_(['Allow_Negative_Balance', 'Allow_Negative_Balances'], 'No'));

  if (!allowNegative && isDebit && endingBalance < 0) {
    logTransaction_(
      student.cardId,
      student.name,
      mode,
      amount,
      startingBalance,
      startingBalance,
      itemId,
      itemName,
      note,
      'Denied - insufficient balance'
    );

    return {
      success: false,
      message: 'Denied: ' + student.name + ' does not have enough balance.'
    };
  }

  updateStudentBalance_(student.row, endingBalance);

  logTransaction_(
    student.cardId,
    student.name,
    mode,
    amount,
    startingBalance,
    endingBalance,
    itemId,
    itemName,
    note,
    'Success'
  );

  return {
    success: true,

function updateAttendanceStreak_(student, todayKey) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(CFG.SHEETS.STUDENTS);
  const map = getFlexibleStudentMap_(sheet);

  if (map.attendanceStreak === -1) {
    throw new Error('Students sheet is missing Attendance_Streak column.');
  }

  const lastDateKey = valueToDateKey_(student.lastAttendanceDate);
  const yesterdayKey = formatDate_(new Date(new Date().getTime() - 24 * 60 * 60 * 1000), 'yyyy-MM-dd');

  let newStreak = 1;

  if (lastDateKey === todayKey) {
    newStreak = Number(student.attendanceStreak) || 1;
  } else if (lastDateKey === yesterdayKey) {
    newStreak = (Number(student.attendanceStreak) || 0) + 1;
  } else {
    newStreak = 1;
  }

  const rowValues = sheet.getRange(student.row, 1, 1, sheet.getLastColumn()).getValues()[0];

  // IMPORTANT: Uses header mapping. In your sheet this should write to column I, not G.
  rowValues[map.attendanceStreak] = newStreak;

  if (map.lastAttendanceDate !== -1) {
    rowValues[map.lastAttendanceDate] = todayKey;
  }

  if (map.lastUpdated !== -1) {
    rowValues[map.lastUpdated] = formatDate_(new Date(), 'yyyy-MM-dd HH:mm:ss');
  }

  sheet.getRange(student.row, 1, 1, rowValues.length).setValues([rowValues]);

  student.attendanceStreak = newStreak;
  student.lastAttendanceDate = todayKey;

  return newStreak;
}

function maybeApplyStreakBonus_(cardId) {
  const updatedStudent = findStudentByCard_(cardId);

  if (!updatedStudent) {
    return {
      amount: 0,
      message: ''
    };
  }

  const bonusDays = Number(getSetting_('Streak_Bonus_Days', 5));
  const bonusAmount = Number(getSetting_('Streak_Bonus_Amount', 0));

  if (!bonusDays || !bonusAmount) {
    return {
      amount: 0,
      message: ''
    };
  }

  const streak = Number(updatedStudent.attendanceStreak) || 0;

  if (streak > 0 && streak % bonusDays === 0) {
    const result = applyBalanceChange_(
      updatedStudent,
      'REWARD',
      bonusAmount,
      streak + '-day attendance streak bonus',
      '',
      ''
    );

    if (result.success) {
      return {
        amount: bonusAmount,
        message: streak + '-day streak bonus applied'
      };
    }
  }

  return {
    amount: 0,
    message: ''
  };
}


function isLate_(dateObj) {
  const cutoff = getLateCutoffTime_();
  const currentHHMM = formatDate_(dateObj, 'HH:mm');

  return timeToMinutes_(currentHHMM) > timeToMinutes_(cutoff);
}



function getAttendanceStatus_(dateObj) {
  const currentHHMM = formatDate_(dateObj, 'HH:mm');

  const startTime = normalizeTimeValue_(
    getSettingWithAliases_(
      ['Attendance_Start_Time'],
      '07:40'
    )
  ) || '07:40';

  const cutoffTime = getLateCutoffTime_();

  const currentMinutes = timeToMinutes_(currentHHMM);
  const startMinutes = timeToMinutes_(startTime);
  const cutoffMinutes = timeToMinutes_(cutoffTime);

  if (currentMinutes < startMinutes) {
    return 'Too Early';
  }

  if (currentMinutes <= cutoffMinutes) {
    return 'Present';
  }

  return 'Late';
}


function getLateCutoffTime_() {
  // IMPORTANT:
  // Do not read a hardcoded cell like Settings!B5 here.
  // If B5 contains a number like 0, Sheets can convert it to 00:00,
  // which makes every scan after midnight count as late.
  const cutoff = getSettingWithAliases_(
    ['Late_Cutoff_Time'],
    '08:35'
  );

  return normalizeTimeValue_(cutoff) || '08:35';
}


function normalizeTimeValue_(value) {
  if (value === null || typeof value === 'undefined' || value === '') return '';

  // If Google returns a Date object for a time cell, format it directly.
  if (Object.prototype.toString.call(value) === '[object Date]' && !isNaN(value)) {
    return Utilities.formatDate(value, CFG.KOREA_TZ, 'HH:mm');
  }

  // If Sheets/Excel serial time is returned as a number, convert fraction of day to HH:mm.
  if (typeof value === 'number' && !isNaN(value)) {
    const fraction = value - Math.floor(value);
    const totalMinutes = Math.round(fraction * 24 * 60);
    const hours = Math.floor(totalMinutes / 60) % 24;
    const minutes = totalMinutes % 60;
    return pad2_(hours) + ':' + pad2_(minutes);
  }

  const text = cleanString_(value);
  if (!text) return '';

  // Serial time stored as text, e.g. "0.3576388888888889".
  if (/^\d*\.?\d+$/.test(text)) {
    const num = Number(text);
    if (!isNaN(num)) {
      const fraction = num - Math.floor(num);
      const totalMinutes = Math.round(fraction * 24 * 60);
      const hours = Math.floor(totalMinutes / 60) % 24;
      const minutes = totalMinutes % 60;
      return pad2_(hours) + ':' + pad2_(minutes);
    }
  }

  // 24-hour text, e.g. "08:35" or "08:35:00".
  let match = text.match(/^(\d{1,2}):(\d{2})(?::\d{2})?$/);
  if (match) {
    const h = Math.max(0, Math.min(23, Number(match[1]) || 0));
    const m = Math.max(0, Math.min(59, Number(match[2]) || 0));
    return pad2_(h) + ':' + pad2_(m);
  }

  // 12-hour text, e.g. "8:35 AM".
  match = text.match(/^(\d{1,2}):(\d{2})(?::\d{2})?\s*(AM|PM)$/i);
  if (match) {
    let h = Number(match[1]) || 0;
    const m = Math.max(0, Math.min(59, Number(match[2]) || 0));
    const ampm = match[3].toUpperCase();

    if (ampm === 'PM' && h < 12) h += 12;
    if (ampm === 'AM' && h === 12) h = 0;

    return pad2_(h) + ':' + pad2_(m);
  }

  return '';
}


function pad2_(num) {
  return String(num).padStart(2, '0');
}


function timeToMinutes_(hhmm) {
  const normalized = normalizeTimeValue_(hhmm);
  const parts = cleanString_(normalized || hhmm).split(':');
  const h = Number(parts[0]) || 0;
  const m = Number(parts[1]) || 0;

  return h * 60 + m;
}

function findStoreItemById_(itemId) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(CFG.SHEETS.STORE);

  if (!sheet) {
    throw new Error('Store sheet not found. Make sure the tab is named exactly "Store".');
  }

  const values = getUsedValues_(sheet);

  if (values.length < 2) {
    throw new Error('Store sheet has no items.');
  }

  const headerInfo = findStoreHeaderRow_(values);

  if (!headerInfo) {
    throw new Error('Could not find Store headers. Store must include Item_ID, Item_Name, and Price.');
  }

  const headerRowIndex = headerInfo.headerRowIndex;
  const headers = headerInfo.headers;

  const itemIdCol = findHeaderIndex_(headers, ['Item_ID', 'Item ID', 'ItemID']);
  const itemNameCol = findHeaderIndex_(headers, ['Item_Name', 'Item Name', 'Name']);
  const priceCol = findHeaderIndex_(headers, ['Price', 'Cost']);
  const activeCol = findHeaderIndex_(headers, ['Active', 'Available']);
  const inventoryCol = findHeaderIndex_(headers, ['Inventory', 'Stock']);
  const categoryCol = findHeaderIndex_(headers, ['Category']);

  if (itemIdCol === -1) throw new Error('Store sheet is missing Item_ID.');
  if (itemNameCol === -1) throw new Error('Store sheet is missing Item_Name.');
  if (priceCol === -1) throw new Error('Store sheet is missing Price.');

  const target = normalizeItemSearch_(itemId);

  for (let i = headerRowIndex + 1; i < values.length; i++) {
    const rowItemId = normalizeItemSearch_(values[i][itemIdCol]);
    const rowItemName = normalizeItemSearch_(values[i][itemNameCol]);

    if (!rowItemId && !rowItemName) continue;

    if (rowItemId === target || rowItemName === target) {
      return {
        row: i + 1,
        id: cleanString_(values[i][itemIdCol]),
        name: cleanString_(values[i][itemNameCol]),
        price: Number(values[i][priceCol]) || 0,
        active: activeCol !== -1 ? values[i][activeCol] : 'Yes',
        inventory: inventoryCol !== -1 ? values[i][inventoryCol] : '',
        inventoryColumn: inventoryCol !== -1 ? inventoryCol + 1 : null,
        category: categoryCol !== -1 ? cleanString_(values[i][categoryCol]) : ''
      };
    }
  }

  return null;
}


function reduceStoreInventory_(item) {
  if (item.inventory === '' || isNaN(Number(item.inventory))) return;
  if (!item.inventoryColumn) return;

  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(CFG.SHEETS.STORE);
  const newInventory = Math.max(0, Number(item.inventory) - 1);

  sheet.getRange(item.row, item.inventoryColumn).setValue(newInventory);
}


function reduceStoreInventoryByQuantity_(item, quantity) {
  if (item.inventory === '' || isNaN(Number(item.inventory))) return;
  if (!item.inventoryColumn) return;

  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(CFG.SHEETS.STORE);
  const newInventory = Math.max(0, Number(item.inventory) - Number(quantity));

  sheet.getRange(item.row, item.inventoryColumn).setValue(newInventory);
}


function updateStudentInventory_(student, item, quantity) {
  const sheet = getOrCreateSheet_(CFG.SHEETS.STUDENT_INVENTORY);

  ensureHeaderRow_(sheet, CFG.HEADERS.STUDENT_INVENTORY, ['Card_ID', 'Item_ID']);

  const values = getUsedValues_(sheet);
  const headerRowIndex = findHeaderRowByNames_(values, ['Card_ID', 'Item_ID']);

  if (headerRowIndex === -1) {
    throw new Error('Student_Inventory headers not found.');
  }

  const headers = values[headerRowIndex];

  const cardCol = findHeaderIndex_(headers, ['Card_ID', 'Card ID']);
  const studentNameCol = findHeaderIndex_(headers, ['Student_Name', 'Student Name']);
  const itemIdCol = findHeaderIndex_(headers, ['Item_ID', 'Item ID']);
  const itemNameCol = findHeaderIndex_(headers, ['Item_Name', 'Item Name']);
  const categoryCol = findHeaderIndex_(headers, ['Category']);
  const quantityCol = findHeaderIndex_(headers, ['Quantity_Purchased', 'Quantity Purchased', 'Quantity']);
  const totalSpentCol = findHeaderIndex_(headers, ['Total_Spent', 'Total Spent']);
  const lastPurchasedCol = findHeaderIndex_(headers, ['Last_Purchased', 'Last Purchased']);
  const lastUpdatedCol = findHeaderIndex_(headers, ['Last_Updated', 'Last Updated']);

  const requiredCols = [cardCol, studentNameCol, itemIdCol, itemNameCol, categoryCol, quantityCol, totalSpentCol, lastPurchasedCol, lastUpdatedCol];

  if (requiredCols.some(function (col) { return col === -1; })) {
    throw new Error('Student_Inventory is missing one or more required headers.');
  }

  const now = formatDate_(new Date(), 'yyyy-MM-dd HH:mm:ss');
  const targetCard = normalizeCardId_(student.cardId);
  const targetItem = normalizeItemSearch_(item.id);
  const itemTotal = (Number(item.price) || 0) * quantity;

  for (let i = headerRowIndex + 1; i < values.length; i++) {
    const rowCard = normalizeCardId_(values[i][cardCol]);
    const rowItem = normalizeItemSearch_(values[i][itemIdCol]);

    if (rowCard === targetCard && rowItem === targetItem) {
      const row = values[i].slice(0, CFG.HEADERS.STUDENT_INVENTORY.length);

      row[cardCol] = student.cardId;
      row[studentNameCol] = student.name;
      row[itemIdCol] = item.id;
      row[itemNameCol] = item.name;
      row[categoryCol] = item.category || '';
      row[quantityCol] = (Number(row[quantityCol]) || 0) + quantity;
      row[totalSpentCol] = (Number(row[totalSpentCol]) || 0) + itemTotal;
      row[lastPurchasedCol] = now;
      row[lastUpdatedCol] = now;

      // One batched row update instead of many individual setValue calls.
      sheet.getRange(i + 1, 1, 1, row.length).setValues([row]);
      return;
    }
  }

  appendRowsToSheet_(sheet, [[
    student.cardId,
    student.name,
    item.id,
    item.name,
    item.category || '',
    quantity,
    itemTotal,
    now,
    now
  ]]);
}


function findStoreHeaderRow_(values) {
  for (let r = 0; r < Math.min(values.length, 10); r++) {
    const normalizedHeaders = values[r].map(function (header) {
      return normalizeHeader_(header);
    });

    const hasItemId = normalizedHeaders.indexOf('item_id') !== -1;
    const hasItemName = normalizedHeaders.indexOf('item_name') !== -1;
    const hasPrice = normalizedHeaders.indexOf('price') !== -1;

    if (hasItemId && hasItemName && hasPrice) {
      return {
        headerRowIndex: r,
        headers: values[r]
      };
    }
  }

  return null;
}


function normalizeItemSearch_(value) {
  return cleanString_(value)
    .toLowerCase()
    .replace(/\s+/g, '')
    .replace(/-/g, '')
    .replace(/_/g, '');
}

function processPayroll_(isAutomatic) {
  ensurePayrollSettings_();
  ensurePayrollLogSheet_();

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const jobsSheet = ss.getSheetByName(CFG.SHEETS.JOBS);

  if (!jobsSheet) {
    throw new Error('Jobs sheet not found.');
  }

  const values = getUsedValues_(jobsSheet);
  const headerRowIndex = findHeaderRowByNames_(values, ['Card_ID', 'Salary']);

  if (headerRowIndex === -1) {
    throw new Error('Jobs sheet headers not found. Required: Card_ID and Salary.');
  }

  const headers = values[headerRowIndex];

  const cardCol = findHeaderIndex_(headers, ['Card_ID', 'Card ID']);
  const nameCol = findHeaderIndex_(headers, ['Student_Name', 'Student Name']);
  const jobTitleCol = findHeaderIndex_(headers, ['Job_Title', 'Job Title', 'Job']);
  const salaryCol = findHeaderIndex_(headers, ['Salary']);
  const payFrequencyCol = findHeaderIndex_(headers, ['Pay_Frequency', 'Pay Frequency']);
  const activeCol = findHeaderIndex_(headers, ['Active']);

  const period = getPayrollPeriod_();

  let paidCount = 0;
  let skippedCount = 0;
  let duplicateCount = 0;
  let totalPaid = 0;

  for (let i = headerRowIndex + 1; i < values.length; i++) {
    const cardId = cleanString_(values[i][cardCol]);
    const salary = Number(values[i][salaryCol]) || 0;
    const active = activeCol !== -1 ? values[i][activeCol] : 'Yes';

    if (!cardId || !salary || !isTruthy_(active)) {
      skippedCount++;
      continue;
    }

    const student = findStudentByCard_(cardId);

    if (!student) {
      logPayrollResult_({
        runAt: period.runAt,
        periodStart: period.periodStart,
        periodEnd: period.periodEnd,
        durationDays: period.durationDays,
        cardId: cardId,
        studentName: nameCol !== -1 ? cleanString_(values[i][nameCol]) : '',
        jobTitle: jobTitleCol !== -1 ? cleanString_(values[i][jobTitleCol]) : '',
        salary: salary,
        payFrequency: payFrequencyCol !== -1 ? cleanString_(values[i][payFrequencyCol]) : '',
        automatic: isAutomatic,
        status: 'Skipped - student not found'
      });

      skippedCount++;
      continue;
    }

    const jobTitle = jobTitleCol !== -1
      ? cleanString_(values[i][jobTitleCol])
      : student.jobTitle || 'Class Job';

    const payFrequency = payFrequencyCol !== -1
      ? cleanString_(values[i][payFrequencyCol])
      : 'Weekly';

    if (wasPayrollAlreadyPaidWithinPeriod_(student.cardId, period.periodStart, period.periodEnd)) {
      logPayrollResult_({
        runAt: period.runAt,
        periodStart: period.periodStart,
        periodEnd: period.periodEnd,
        durationDays: period.durationDays,
        cardId: student.cardId,
        studentName: student.name,
        jobTitle: jobTitle,
        salary: salary,
        payFrequency: payFrequency,
        automatic: isAutomatic,
        status: 'Duplicate blocked'
      });

      duplicateCount++;
      continue;
    }

    const result = applyBalanceChange_(
      student,
      'PAYROLL',
      salary,
      'Job payroll: ' + jobTitle + ' | Pay period: ' + period.periodStart + ' to ' + period.periodEnd,
      '',
      ''
    );

    if (result.success) {
      logPayrollResult_({
        runAt: period.runAt,
        periodStart: period.periodStart,
        periodEnd: period.periodEnd,
        durationDays: period.durationDays,
        cardId: student.cardId,
        studentName: student.name,
        jobTitle: jobTitle,
        salary: salary,
        payFrequency: payFrequency,
        automatic: isAutomatic,
        status: 'Paid'
      });

      paidCount++;
      totalPaid += salary;
    } else {
      logPayrollResult_({
        runAt: period.runAt,
        periodStart: period.periodStart,
        periodEnd: period.periodEnd,
        durationDays: period.durationDays,
        cardId: student.cardId,
        studentName: student.name,
        jobTitle: jobTitle,
        salary: salary,
        payFrequency: payFrequency,
        automatic: isAutomatic,
        status: 'Skipped - ' + result.message
      });

      skippedCount++;
    }
  }

  return {
    runAt: period.runAt,
    periodStart: period.periodStart,
    periodEnd: period.periodEnd,
    durationDays: period.durationDays,
    paidCount: paidCount,
    skippedCount: skippedCount,
    duplicateCount: duplicateCount,
    totalPaid: totalPaid
  };
}

  });
}

function getPayrollPeriod_() {
  const now = new Date();
  const runAt = formatDate_(now, 'yyyy-MM-dd HH:mm:ss');
  const todayKey = formatDate_(now, 'yyyy-MM-dd');

  // Apps Script formatDate pattern "u": Monday = 1, Tuesday = 2, ..., Sunday = 7.
  const dayOfWeek = Number(formatDate_(now, 'u'));
  const daysSinceMonday = dayOfWeek - 1;
  const mondayDate = addDaysToDateKey_(todayKey, -daysSinceMonday);
  const fridayDate = addDaysToDateKey_(mondayDate, 4);

  return {
    runAt: runAt,
    periodStart: mondayDate,
    periodEnd: fridayDate,
    durationDays: 5
  };
}


function addDaysToDateKey_(dateKey, daysToAdd) {
  const parts = String(dateKey).split('-');

  const year = Number(parts[0]);
  const month = Number(parts[1]) - 1;
  const day = Number(parts[2]);

  const date = new Date(year, month, day);
  date.setDate(date.getDate() + daysToAdd);

  return formatDate_(date, 'yyyy-MM-dd');
}

 GOOGLE DOC ATTENDANCE REPORT

function generateAttendanceGoogleDocReport() {
  const ui = SpreadsheetApp.getUi();

  const datePrompt = ui.prompt(
    'Attendance Report Date',
    'Enter date as yyyy-mm-dd. Leave blank for today.',
    ui.ButtonSet.OK_CANCEL
  );

  if (datePrompt.getSelectedButton() !== ui.Button.OK) return;

  let reportDate = cleanString_(datePrompt.getResponseText());
  if (!reportDate) reportDate = formatDate_(new Date(), 'yyyy-MM-dd');

  const classPrompt = ui.prompt(
    'Class / Period',
    'Enter class/period exactly, or leave blank for all classes.',
    ui.ButtonSet.OK_CANCEL
  );

  if (classPrompt.getSelectedButton() !== ui.Button.OK) return;

  const classPeriod = cleanString_(classPrompt.getResponseText());

  const doc = DocumentApp.create('Attendance Report - ' + reportDate + (classPeriod ? ' - ' + classPeriod : ''));
  const body = doc.getBody();

  body.appendParagraph('Attendance Report').setHeading(DocumentApp.ParagraphHeading.HEADING1);
  body.appendParagraph('Date: ' + reportDate);
  body.appendParagraph('Class / Period: ' + (classPeriod || 'All classes'));
  body.appendParagraph('Timezone: Asia/Seoul');
  body.appendParagraph('');

  const students = getAllStudents_();
  const attendance = getDailyAttendanceRecords_(reportDate, classPeriod);

  const present = [];
  const late = [];
  const absent = [];

  const attendanceByCard = {};

  attendance.forEach(function (record) {
    attendanceByCard[normalizeCardId_(record.cardId)] = record;
  });

  students.forEach(function (student) {
    const record = attendanceByCard[normalizeCardId_(student.cardId)];

    if (!record) {
      absent.push(student.name);
    } else if (record.status === 'Late') {
      late.push(student.name + ' - ' + record.time);
    } else if (record.status === 'Present') {
      present.push(student.name + ' - ' + record.time);
    } else {
      absent.push(student.name + ' - ' + record.status);
    }
  });

  body.appendParagraph('Summary').setHeading(DocumentApp.ParagraphHeading.HEADING2);
  body.appendParagraph('Present: ' + present.length);
  body.appendParagraph('Late: ' + late.length);
  body.appendParagraph('Absent / No Scan: ' + absent.length);
  body.appendParagraph('');

  appendListSection_(body, 'Present', present);
  appendListSection_(body, 'Late', late);
  appendListSection_(body, 'Absent / No Scan', absent);

  doc.saveAndClose();

  const url = doc.getUrl();
  writeMainResult_('Attendance report created: ' + url);
  ui.alert('Attendance report created:\n\n' + url);
}


function getDailyAttendanceRecords_(dateKey, classPeriod) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(CFG.SHEETS.DAILY_ATTENDANCE);
  const values = getUsedValues_(sheet);
  const headerRowIndex = findHeaderRowByNames_(values, ['Date', 'Card_ID', 'Status']);

  if (headerRowIndex === -1) return [];

  const headers = values[headerRowIndex];
  const dateCol = findHeaderIndex_(headers, ['Date']);
  const cardCol = findHeaderIndex_(headers, ['Card_ID', 'Card ID']);
  const nameCol = findHeaderIndex_(headers, ['Student_Name', 'Student Name']);
  const classCol = findHeaderIndex_(headers, ['Class_Period', 'Class Period']);
  const statusCol = findHeaderIndex_(headers, ['Status']);
  const timeCol = findHeaderIndex_(headers, ['Check_In_Time', 'Check In Time']);

  const records = [];

  for (let i = headerRowIndex + 1; i < values.length; i++) {
    const rowDate = valueToDateKey_(values[i][dateCol]);
    const rowClass = cleanString_(values[i][classCol]);

    if (rowDate !== dateKey) continue;
    if (classPeriod && rowClass !== classPeriod) continue;

    records.push({
      date: rowDate,
      cardId: cleanString_(values[i][cardCol]),
      name: cleanString_(values[i][nameCol]),
      classPeriod: rowClass,
      status: cleanString_(values[i][statusCol]),
      time: cleanString_(values[i][timeCol])
    });
  }

  return records;
}

  });

  body.appendParagraph('');
}

function findStudentByCard_(cardId) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(CFG.SHEETS.STUDENTS);
  const lastRow = sheet.getLastRow();

  if (lastRow < 2) return null;

  const map = getFlexibleStudentMap_(sheet);

  if (map.cardId === -1 || map.studentName === -1) {
    throw new Error('Students sheet must include Card_ID and Student_Name headers.');
  }

  const cleanCard = cleanString_(cardId);
  const targetCard = normalizeCardId_(cleanCard);
  const lastCol = sheet.getLastColumn();

  // Fast path: search only the mapped Card_ID column.
  const cardRange = sheet.getRange(map.headerRowIndex + 2, map.cardId + 1, lastRow - map.headerRowIndex - 1, 1);
  let foundCell = null;

  try {
    foundCell = cardRange
      .createTextFinder(cleanCard)
      .matchEntireCell(true)
      .matchCase(false)
      .findNext();
  } catch (err) {
    foundCell = null;
  }

  if (foundCell) {
    const row = foundCell.getRow();
    const values = sheet.getRange(row, 1, 1, lastCol).getValues()[0];
    return buildStudentFromRow_(row, values, map);
  }

  // Fallback: normalized matching for scanners/cards with hidden characters or formatting issues.
  const values = sheet.getRange(map.headerRowIndex + 2, 1, lastRow - map.headerRowIndex - 1, lastCol).getValues();

  for (let i = 0; i < values.length; i++) {
    const rowCard = normalizeCardId_(values[i][map.cardId]);

    if (rowCard === targetCard) {
      return buildStudentFromRow_(map.headerRowIndex + 2 + i, values[i], map);
    }
  }

  return null;
}


function buildStudentFromRow_(row, values, map) {
  return {
    row: row,
    cardId: map.cardId !== -1 ? cleanString_(values[map.cardId]) : '',
    name: map.studentName !== -1 ? cleanString_(values[map.studentName]) : '',
    grade: map.grade !== -1 ? values[map.grade] : '',
    homeroom: map.homeroom !== -1 ? values[map.homeroom] : '',
    balance: map.balance !== -1 ? Number(values[map.balance]) || 0 : 0,
    jobTitle: map.jobTitle !== -1 ? cleanString_(values[map.jobTitle]) : '',
    active: map.active !== -1 ? values[map.active] : 'Yes',
    lastUpdated: map.lastUpdated !== -1 ? values[map.lastUpdated] : '',
    attendanceStreak: map.attendanceStreak !== -1 ? Number(values[map.attendanceStreak]) || 0 : 0,
    lastAttendanceDate: map.lastAttendanceDate !== -1 ? values[map.lastAttendanceDate] : '',
    notes: map.notes !== -1 ? values[map.notes] : ''
  };
}

function getSetting_(key, defaultValue) {
  const cache = CacheService.getDocumentCache();
  const cacheKey = 'SETTINGS_MAP_V1';
  let settingsMap = null;

  if (cache) {
    const cached = cache.get(cacheKey);
    if (cached) {
      try {
        settingsMap = JSON.parse(cached);
      } catch (err) {
        settingsMap = null;
      }
    }
  }

  if (!settingsMap) {
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(CFG.SHEETS.SETTINGS);
    if (!sheet) return defaultValue;

    const values = getUsedValues_(sheet);
    settingsMap = {};

    for (let i = 1; i < values.length; i++) {
      const settingKey = cleanString_(values[i][0]);
      if (settingKey) settingsMap[settingKey] = values[i][1];
    }

    if (cache) {
      cache.put(cacheKey, JSON.stringify(settingsMap), 300);
    }
  }

  const value = settingsMap[key];

  if (value === '' || value === null || typeof value === 'undefined') {
    return defaultValue;
  }

  return value;
}


function getSettingWithAliases_(keys, defaultValue) {
  const missing = '__SETTING_MISSING__';

  for (let i = 0; i < keys.length; i++) {
    const value = getSetting_(keys[i], missing);

    if (value !== missing) {
      return value;
    }
  }

  return defaultValue;
}

function registerProfileScanAttempt_() {
  const nowMs = new Date().getTime();
  let attempts = getProfileSecurityProperty_('attempts', '');

  if (!attempts) {
    attempts = [];
  } else {
    try {
      attempts = JSON.parse(attempts);
    } catch (err) {
      attempts = [];
    }
  }

  const oneMinuteAgo = nowMs - 60 * 1000;

  attempts = attempts.filter(function (timestamp) {
    return Number(timestamp) >= oneMinuteAgo;
  });

  attempts.push(nowMs);
  setProfileSecurityProperty_('attempts', JSON.stringify(attempts));
}


function checkAndApplyProfileLock_(sheet) {
  let attempts = getProfileSecurityProperty_('attempts', '');

  if (!attempts) {
    attempts = [];
  } else {
    try {
      attempts = JSON.parse(attempts);
    } catch (err) {
      attempts = [];
    }
  }

  const nowMs = new Date().getTime();
  const oneMinuteAgo = nowMs - 60 * 1000;

  attempts = attempts.filter(function (timestamp) {
    return Number(timestamp) >= oneMinuteAgo;
  });

  if (attempts.length > PROFILE_SECURITY.MAX_ATTEMPTS_PER_MINUTE) {
    const lockUntilMs = nowMs + PROFILE_SECURITY.LOCK_MINUTES * 60 * 1000;

    setProfileSecurityProperty_('lockUntilMs', String(lockUntilMs));
    setProfileSecurityProperty_('attempts', JSON.stringify([]));

    setStatus_(
      sheet,
      'Locked. More than ' +
      PROFILE_SECURITY.MAX_ATTEMPTS_PER_MINUTE +
      ' attempts in 1 minute. Try again in ' +
      PROFILE_SECURITY.LOCK_MINUTES +
      ' minutes.'
    );

    return true;
  }

  setProfileSecurityProperty_('attempts', JSON.stringify(attempts));
  return false;
}

  const props = PropertiesService.getScriptProperties();

  return value;
}

function getUsedValues_(sheet) {
  if (!sheet) return [];

  const lastRow = sheet.getLastRow();
  const lastCol = sheet.getLastColumn();

  if (lastRow < 1 || lastCol < 1) return [];

  // V2.19 SAFE PATCH:
  // Replacement for getDataRange().getValues().
  // This avoids reading thousands of empty/formatted rows.
  return sheet.getRange(1, 1, lastRow, lastCol).getValues();
}


function ensureRows_(sheet, rowsNeeded) {
  if (!sheet || !rowsNeeded) return;

  const maxRows = sheet.getMaxRows();
  const lastRow = Math.max(sheet.getLastRow(), 1);
  const availableBlankRows = maxRows - lastRow;

  if (availableBlankRows < rowsNeeded) {
    sheet.insertRowsAfter(maxRows, rowsNeeded - availableBlankRows);
  }
}


function getColumnByHeader_(sheet, headerName) {
  if (!sheet) throw new Error('Missing sheet for header lookup.');

  const headers = sheet
    .getRange(1, 1, 1, sheet.getLastColumn())
    .getValues()[0];

  const index = headers.indexOf(headerName);

  if (index === -1) {
    throw new Error('Missing header: ' + headerName + ' on sheet ' + sheet.getName());
  }

  return index + 1;
}

function ensureHeaderRow_(sheet, headers, requiredNames) {
  const values = getUsedValues_(sheet);
  let headerRowIndex = findHeaderRowByNames_(values, requiredNames || headers);

  if (headerRowIndex === -1) {
    headerRowIndex = 0;
  }

  sheet.getRange(headerRowIndex + 1, 1, 1, headers.length).setValues([headers]);

function findHeaderRowByNames_(values, requiredNames) {
  if (!values || !values.length) return -1;

  for (let r = 0; r < Math.min(values.length, 10); r++) {
    const normalizedHeaders = values[r].map(function (header) {
      return normalizeHeader_(header);
    });

    let foundCount = 0;

    requiredNames.forEach(function (name) {
      const target = normalizeHeader_(name);

      if (normalizedHeaders.indexOf(target) !== -1) {
        foundCount++;
      }
    });

    if (foundCount >= Math.min(requiredNames.length, 2)) {
      return r;
    }
  }

  return -1;
}


function findHeaderIndex_(headers, possibleNames) {
  const normalizedHeaders = headers.map(function (header) {
    return normalizeHeader_(header);
  });

  for (let i = 0; i < possibleNames.length; i++) {
    const target = normalizeHeader_(possibleNames[i]);
    const index = normalizedHeaders.indexOf(target);

    if (index !== -1) {
      return index;
    }
  }

  return -1;
}

      if (headerRowIndex === -1) headerRowIndex = 0;
    }

    sheet.getRange(headerRowIndex + 1, 1, 1, Math.max(sheet.getLastColumn(), 1))
      .setFontWeight('bold')
      .setBackground('#e8f0fe');

    sheet.setFrozenRows(headerRowIndex + 1);
  });
}


function applyPlainTextFormats_() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();

  const students = ss.getSheetByName(CFG.SHEETS.STUDENTS);
  if (students) students.getRange('A:A').setNumberFormat('@');

  const store = ss.getSheetByName(CFG.SHEETS.STORE);

  if (inventory) {

    inventory.getRange('C:C').setNumberFormat('@');
  }
}




function appendRowsToSheet_(sheet, rows) {
  if (!sheet || !rows || !rows.length) return;

  const cleanedRows = rows.filter(function (row) {
    return row && row.length;
  });

  if (!cleanedRows.length) return;

  const startRow = sheet.getLastRow() + 1;
  const numRows = cleanedRows.length;
  const numCols = cleanedRows[0].length;

  // V2.19 SAFE PATCH:
  // If unused rows were deleted, create enough new rows before writing.
  ensureRows_(sheet, numRows + 10);

  // Fast batched write. Do not use appendRow() for transaction/log rows.
  sheet.getRange(startRow, 1, numRows, numCols).setValues(cleanedRows);
}




/*******************************************************
 FAST ATTENDANCE INDEX HELPERS
 Avoids reading the entire Daily_Attendance sheet on every scan.
*******************************************************/

function buildAttendanceIndexKey_(cardId, dateKey, classPeriod) {
  const cleanDate = valueToDateKey_(dateKey);
  const cleanCard = normalizeCardId_(cardId);
  const cleanClass = cleanString_(classPeriod).toLowerCase().replace(/[^a-z0-9]/g, '_');

  return 'ATT_IDX_' + cleanDate + '_' + cleanCard + '_' + cleanClass;
}


function markAttendanceRecorded_(cardId, dateKey, classPeriod) {
  const key = buildAttendanceIndexKey_(cardId, dateKey, classPeriod);
  const cache = CacheService.getDocumentCache();
  const props = PropertiesService.getDocumentProperties();

  props.setProperty(key, '1');
  if (cache) cache.put(key, '1', 21600);
}


function rebuildTodayAttendanceIndex_() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(CFG.SHEETS.DAILY_ATTENDANCE);

  if (!sheet || sheet.getLastRow() < 2) return;

  const todayKey = formatDate_(new Date(), 'yyyy-MM-dd');
  const values = getUsedValues_(sheet);
  const headerRowIndex = findHeaderRowByNames_(values, ['Date', 'Card_ID', 'Class_Period']);

  if (headerRowIndex === -1) return;

  const headers = values[headerRowIndex];
  const dateCol = findHeaderIndex_(headers, ['Date']);
  const cardCol = findHeaderIndex_(headers, ['Card_ID', 'Card ID']);
  const classCol = findHeaderIndex_(headers, ['Class_Period', 'Class Period']);

  if (dateCol === -1 || cardCol === -1 || classCol === -1) return;

  for (let i = headerRowIndex + 1; i < values.length; i++) {
    const rowDate = valueToDateKey_(values[i][dateCol]);
    const rowCard = cleanString_(values[i][cardCol]);
    const rowClass = cleanString_(values[i][classCol]);

    if (rowDate === todayKey && rowCard && rowClass) {
      markAttendanceRecorded_(rowCard, rowDate, rowClass);
    }
  }
}

/*******************************************************
 GENERIC HELPERS
*******************************************************/

function cleanString_(value) {
  if (value === null || typeof value === 'undefined') return '';
  return String(value).trim();
}


function normalizeHeader_(value) {
  return cleanString_(value)
    .toLowerCase()
    .replace(/\s+/g, '_')
    .replace(/[^a-z0-9_]/g, '');
}


function normalizeCardId_(value) {
  return cleanString_(value)
    .toLowerCase()
    .replace(/[\u200B-\u200D\uFEFF]/g, '') // invisible chars
    .replace(/\s+/g, '')                  // spaces/tabs/newlines
    .replace(/[^a-z0-9]/g, '');           // only letters/numbers
}


function isTruthy_(value) {
  const text = cleanString_(value).toLowerCase();

  return ['yes', 'y', 'true', '1', 'active', 'available'].indexOf(text) !== -1;
}


function getTimeZone_() {
  return CFG.KOREA_TZ;
}


function formatDate_(dateObj, pattern) {
  return Utilities.formatDate(dateObj, getTimeZone_(), pattern);
}


function valueToDateKey_(value) {
  if (!value) return '';

  if (Object.prototype.toString.call(value) === '[object Date]' && !isNaN(value)) {
    return formatDate_(value, 'yyyy-MM-dd');
  }

  const text = cleanString_(value);

  if (/^\d{4}-\d{2}-\d{2}$/.test(text)) return text;

  const parsed = new Date(text);

  if (!isNaN(parsed)) {
    return formatDate_(parsed, 'yyyy-MM-dd');
  }

  return text;
}


function dateKeyToNumber_(dateKey) {
  const clean = valueToDateKey_(dateKey);
  const parts = clean.split('-');

  if (parts.length !== 3) return 0;

  return Number(parts[0]) * 10000 + Number(parts[1]) * 100 + Number(parts[2]);
}


function dateRangesOverlap_(startA, endA, startB, endB) {
  const aStart = dateKeyToNumber_(startA);
  const aEnd = dateKeyToNumber_(endA);
  const bStart = dateKeyToNumber_(startB);
  const bEnd = dateKeyToNumber_(endB);

  if (!aStart || !aEnd || !bStart || !bEnd) return false;

  return aStart <= bEnd && bStart <= aEnd;
}
